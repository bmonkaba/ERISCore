37 void erisAudioAnalyzeScope::update(void)
38 {
39  audio_block_t *block;
40  audio_block_t *blockb; //2nd channel (optional)
41  bool isDualChannel;
42 
43  uint32_t offset;
44  uint32_t remain;
45 
46  //capture only if channel 0 is connected
47  block = receiveReadOnly(0);
48  if (!block){
49  blockb = receiveReadOnly(1);
50  if (blockb) release(blockb);
51  return;
52  }
53 
54  blockb = receiveReadOnly(1);
55  if (!blockb) {
56  isDualChannel = false;
57  } else isDualChannel = true;
58 
59  offset = 0;
60  while (offset < AUDIO_BLOCK_SAMPLES) {
61  remain = AUDIO_BLOCK_SAMPLES - offset;
62  switch (state) {
63  case STATE_WAIT_TRIGGER:
64  // TODO: implement this....
65  offset = AUDIO_BLOCK_SAMPLES;
66  break;
67 
68  case STATE_DELAY:
69  //Serial.printf("STATE_DELAY, count = %u\n", count);
70  if (remain < count) {
71  count -= remain;
72  offset = AUDIO_BLOCK_SAMPLES;
73  } else {
74  offset += count;
75  count = mem_length;
76  state = STATE_PRINTING;
77  }
78  break;
79 
80  case STATE_PRINTING:
81  //for the first sample try to find a zero crossing offset in the curernt audio block
82  if (count == mem_length){
83  bool found = false;
84  for(int16_t i=1;i < AUDIO_BLOCK_SAMPLES-h_div;i++){
85  if (block->data[i] > 0 && block->data[i-1] < 0){
86  offset = i;
87  found = true;
88  break;
89  }
90  }
91  if(!found) {offset = AUDIO_BLOCK_SAMPLES;};
92  edgeCount = 0;
93  peakValue = 0;
94  }
95 
96  while ((offset < AUDIO_BLOCK_SAMPLES) && (count > 0)){
97  h_div_count++;
98  if(h_div_count==h_div){
99  h_div_count=0;
100  count--;
101  memory[0][mem_length - count - 1 ] = block->data[offset];
102  peakValue = max(peakValue,abs(block->data[offset]));
103  if (isDualChannel){
104  memory[1][mem_length - count - 1 ] = blockb->data[offset];
105  peakValue = max(peakValue,abs(blockb->data[offset]));
106  }else memory[1][mem_length - count - 1 ] = 0;
107 
108  if ((offset >= h_div) && (block->data[offset] >= 0 ) && (block->data[offset-h_div] < 0)) edgeCount++;
109  }
110  offset++;
111  }
112 
113 
114  if (count == 0){
115  //Serial.println(edgeCount);
116  if((edgeCount < 3) && (auto_h_div < 4)) auto_h_div++;
117  if((edgeCount > 4) && (auto_h_div > 1)) auto_h_div--;
118  //calculate the dot product if dual channel
119  if (isDualChannel){
120  dotLast = dot;
121  arm_dot_prod_q15(&memory[0][0],&memory[1][0],mem_length,&dot);
122  dotAvg = (dotAvg*0.9) + (dot*0.1);
123  }
124  isAvailable = true;
125  if (autoTrigger){
126  trigger();
127  }else state = STATE_IDLE;
128  }
129  break;
130 
131  default: // STATE_IDLE
132  offset = AUDIO_BLOCK_SAMPLES;
133  break;
134  }
135  }
136  release(block);
137  if(isDualChannel) release(blockb);
138 }
