6 // Approximates atan2(y, x) normalized to the [0,4] range
7 // with a maximum error of 0.1620 degrees
8 float normalized_atan2( float y, float x )
9 {
10  //return 0;
11  static const uint32_t sign_mask = 0x80000000;
12  static const float b = 0.596227f;
13 
14  // Extract the sign bits
15  uint32_t ux_s = sign_mask & (uint32_t &)x;
16  uint32_t uy_s = sign_mask & (uint32_t &)y;
17 
18  // Determine the quadrant offset
19  float q = (float)( ( ~ux_s & uy_s ) >> 29 | ux_s >> 30 );
20 
21  // Calculate the arctangent in the first quadrant
22  float bxy_a = ::fabs( b * x * y );
23  float num = bxy_a + y * y;
24  float atan_1q = num / ( x * x + bxy_a + num );
25 
26  // Translate it to the proper quadrant
27  uint32_t uatan_2q = (ux_s ^ uy_s) | (uint32_t &)atan_1q;
28  return q + (float &)uatan_2q;
29 }
