50 class AudioSynthWaveform : public AudioStream
51 {
52 public:
53  AudioSynthWaveform(void) : AudioStream(0,NULL),
54  phase_accumulator(0), phase_increment(0), phase_offset(0),
55  magnitude(0), pulse_width(0x40000000),
56  arbdata(NULL), sample(0), tone_type(WAVEFORM_SINE),
57  tone_offset(0) {
58  }
59 
60  void frequency(float freq) {
61  if (freq < 0.0) {
62  freq = 0.0;
63  } else if (freq > AUDIO_SAMPLE_RATE_EXACT / 2) {
64  freq = AUDIO_SAMPLE_RATE_EXACT / 2;
65  }
66  phase_increment = freq * (4294967296.0 / AUDIO_SAMPLE_RATE_EXACT);
67  if (phase_increment > 0x7FFE0000u) phase_increment = 0x7FFE0000;
68  }
69  void phase(float angle) {
70  if (angle < 0.0) {
71  angle = 0.0;
72  } else if (angle > 360.0) {
73  angle = angle - 360.0;
74  if (angle >= 360.0) return;
75  }
76  phase_offset = angle * (4294967296.0 / 360.0);
77  }
78  void amplitude(float n) { // 0 to 1.0
79  if (n < 0) {
80  n = 0;
81  } else if (n > 1.0) {
82  n = 1.0;
83  }
84  magnitude = n * 65536.0;
85  }
86  void offset(float n) {
87  if (n < -1.0) {
88  n = -1.0;
89  } else if (n > 1.0) {
90  n = 1.0;
91  }
92  tone_offset = n * 32767.0;
93  }
94  void pulseWidth(float n) { // 0.0 to 1.0
95  if (n < 0) {
96  n = 0;
97  } else if (n > 1.0) {
98  n = 1.0;
99  }
100  pulse_width = n * 4294967296.0;
101  }
102  void begin(short t_type) {
103  phase_offset = 0;
104  tone_type = t_type;
105  }
106  void begin(float t_amp, float t_freq, short t_type) {
107  amplitude(t_amp);
108  frequency(t_freq);
109  phase_offset = 0;
110  tone_type = t_type;
111  }
112  void arbitraryWaveform(const int16_t *data, float maxFreq) {
113  arbdata = data;
114  }
115  virtual void update(void);
116 
117 private:
118  uint32_t phase_accumulator;
119  uint32_t phase_increment;
120  uint32_t phase_offset;
121  int32_t magnitude;
122  uint32_t pulse_width;
123  const int16_t *arbdata;
124  int16_t sample; // for WAVEFORM_SAMPLE_HOLD
125  short tone_type;
126  int16_t tone_offset;
127 };
