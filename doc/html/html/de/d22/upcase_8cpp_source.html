212 uint16_t toUpcase(uint16_t chr) {
213  uint16_t i, first;
214  // Optimize for simple ASCII.
215  if (chr < 127) {
216  return chr - ('a' <= chr && chr <= 'z' ? 'a' - 'A' : 0);
217  }
218  i = searchPair16(reinterpret_cast<const pair16_t*>(mapTable), MAP_DIM, chr);
219  first = readTable16(mapTable[i].base);
220  if (first <= chr && (chr - first) < readTable8(mapTable[i].count)) {
221  int8_t off = readTable8(mapTable[i].off);
222  if (off == 1) {
223  return chr - ((chr - first) & 1);
224  }
225  return chr + (off ? off : -0x1C60);
226  }
227  i = searchPair16(lookupTable, LOOKUP_DIM, chr);
228  if (readTable16(lookupTable[i].key) == chr) {
229  return readTable16(lookupTable[i].val);
230  }
231  return chr;
232 }
