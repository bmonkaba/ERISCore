162  Spi* pSpi = SPI0;

164 #if USE_SAM3X_DMAC

166  while (pSpi->SPI_SR & (SPI_SR_OVRES | SPI_SR_RDRF)) {pSpi->SPI_RDR;}

174  dmac_channel_disable(SPI_DMAC_TX_CH);
