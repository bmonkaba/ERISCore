36 // Approximates atan2(y, x) normalized to the [0,4] range
37 // with a maximum error of 0.1620 degrees
38 float normalized_atan2( float y, float x )
39 {
40  //return 0;
41  static const uint32_t sign_mask = 0x80000000;
42  static const float b = 0.596227f;
43 
44  // Extract the sign bits
45  uint32_t ux_s = sign_mask & (uint32_t &)x;
46  uint32_t uy_s = sign_mask & (uint32_t &)y;
47 
48  // Determine the quadrant offset
49  float q = (float)( ( ~ux_s & uy_s ) >> 29 | ux_s >> 30 );
50 
51  // Calculate the arctangent in the first quadrant
52  float bxy_a = ::fabs( b * x * y );
53  float num = bxy_a + y * y;
54  float atan_1q = num / ( x * x + bxy_a + num );
55 
56  // Translate it to the proper quadrant
57  uint32_t uatan_2q = (ux_s ^ uy_s) | (uint32_t &)atan_1q;
58  return q + (float &)uatan_2q;
59 }

88 {
