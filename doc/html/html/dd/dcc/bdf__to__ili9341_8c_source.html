64 int pixel(const glyph_t *g, int x, int y)
65 {
66  uint8_t b;
67 
68  if (x >= g->width) return 0;
69  if (y >= g->height) return 0;
70  b = g->data[(((g->width + 7) >> 3) * y) + (x >> 3)];
71  if (b & (1 << (7 - (x & 7)))) return 1;
72  return 0;
73 }

108 void output_bit(int bit)
109 {
110  uint8_t bitmask = 1 << (7 - output_state_bitcount);
111 
112  if (bit) {
113  //printf("(1)");
114  output_state_byte |= bitmask;
115  } else {
116  //printf("(0)");
117  output_state_byte &= ~bitmask;
118  }
119  output_state_bitcount++;
120  if (output_state_bitcount >= 8) {
121  output_state_bitcount = 0;
122  printf("0x%02X,", output_state_byte);
123  output_state_bytecount++;
124  if (++output_state_linelen >= 10) output_newline();
125  output_state_byte = 0;

178 void output_glyph(const glyph_t *g)
179 {
180  int y = 0;
181  int identical_count;
182 
183  //printf("output glyph:");
184  if (ascii_art) print_glyph(g);
185  output_number(0, 3); // reserved bits, intended to identify future formats
186  output_number(g->width, bits_width);
187  output_number(g->height, bits_height);
188  output_number(g->xoffset, bits_xoffset);
189  output_number(g->yoffset, bits_yoffset);
190  output_number(g->delta, bits_delta);
191 
192  while (y < g->height) {
193  //printf("line %d: ", y);
194  identical_count = num_lines_identical(g, y);
195  if (identical_count == 0) {
196  //printf("unique\n");
197  output_bit(0);
198  output_line(g, y);
199  } else {
200  output_bit(1);
201  if (identical_count > 6) identical_count = 6;
202  //printf("duplicate %d times\n", identical_count);
203  output_number(identical_count - 1, 3);
204  output_line(g, y);
205  y += identical_count;
206  }
207  y++;
208  }
209  output_pad_to_byte();
210  if (ascii_art) output_newline();
211 }

293 void parse_bdf(FILE *fp, glyph_t *g)
294 {
295  char line[1024], name[1024], *src, *dst;
296  const char *p;
297  int state = 0, linenum = 0;
298  int found_encoding = -1, found_dwidth = -1, found_bbx = -1;
299  int found_ascent = 0, found_descent = 0;
300  int expect_lines, expect_bytes;
301  int encoding, dwidth_x, dwidth_y, bbx_width, bbx_height, bbx_xoffset, bbx_yoffset;
302  int size, font_ascent = 0, font_descent = 0;
303  uint8_t *data;
304  int i, j;
305 
306  while (fgets(line, sizeof(line), fp)) {
307  linenum++;
308  if (state == 0) {
309  if (strncmp(line, "STARTCHAR", 10) == 0) {
310  //printf("startchar\n");
311  found_encoding = 0;
312  found_dwidth = 0;
313  found_bbx = 0;
314  continue;
315  }
316  if (sscanf(line, "SIZE %d %d %d", &size, &i, &j) == 3) {
317  font_size = size;
318  }
319  if (sscanf(line, "FAMILY_NAME \"%[^\"]\"", name) == 1) {
320  //printf("// name = %s\n", name);
321  src = name;
322  dst = font_name;
323  while (*src) {
324  if (isalnum(*src)) *dst++ = *src;
325  src++;
326  }
327  *dst = 0;
328  }
329  if (strncmp(line, "WEIGHT_NAME \"Bold\"", 18) == 0) {
330  is_bold = 1;
331  }
332  if (strncmp(line, "SLANT \"I\"", 9) == 0) {
333  is_italic = 1;
334  }
335  if (sscanf(line, "FONT_ASCENT %d", &font_ascent) == 1) {
336  found_ascent = 1;
337  }
338  if (sscanf(line, "FONT_DESCENT %d", &font_descent) == 1) {
339  found_descent = 1;
340  }
341  if (sscanf(line, "ENCODING %d", &encoding) == 1) {
342  //printf("encoding %d\n", encoding);
343  found_encoding = 1;
344  #ifdef ENCODING_OFFSET
345  if (encoding >= ENCODING_OFFSET + ENCODING_START
346  && encoding <= ENCODING_OFFSET + ENCODING_END) {
347  encoding -= ENCODING_OFFSET;
348  } else {
349  encoding = ENCODING_OFFSET + ENCODING_END + 1;
350  }
351  #endif
352  continue;
353  }
354  if (sscanf(line, "DWIDTH %d %d", &dwidth_x, &dwidth_y) == 2) {
355  //printf("dwidth %d %d\n", dwidth_x, dwidth_y);
356  if (dwidth_x < 0) {
357  die("DWIDTH x negative, at line %d", linenum);
358  }
359  if (dwidth_y != 0) {
360  die("DWIDTH y not zero, at line %d", linenum);
361  }
362  found_dwidth = 1;
363  continue;
364  }
365  if (sscanf(line, "BBX %d %d %d %d", &bbx_width,
366  &bbx_height, &bbx_xoffset, &bbx_yoffset) == 4) {
367  //printf("bbx %d %d %d %d\n",
368  //bbx_width, bbx_height, bbx_xoffset, bbx_yoffset);
369  if (bbx_width < 0) die("BBX width negative, line %d", linenum);
370  if (bbx_height < 0) die("BBX height negative, line %d", linenum);
371  found_bbx = 1;
372  continue;
373  }
374  if (strncmp(line, "BITMAP", 6) == 0) {
375  if (found_encoding != 1) die("missing ENCODING, line %d", linenum);
376  if (found_dwidth != 1) die("missing DWIDTH, line %d", linenum);
377  if (found_bbx != 1) die("missing BBX, line %d", linenum);
378  expect_lines = bbx_height;
379  expect_bytes = (bbx_width + 7) >> 3;
380  if (encoding >= ENCODING_START && encoding <= ENCODING_END) {
381  data = malloc(expect_lines * expect_bytes);
382  if (!data) die("unable to allocate memory!");
383  g[encoding].width = bbx_width;
384  g[encoding].height = bbx_height;
385  g[encoding].xoffset = bbx_xoffset;
386  g[encoding].yoffset = bbx_yoffset;
387  g[encoding].delta = dwidth_x;
388  g[encoding].encoding = encoding;
389  g[encoding].data = data;
390  } else {
391  data = NULL;
392  }
393  state = 1;
394  continue;
395  }
396  } else {
397  if (expect_lines > 0 && expect_bytes > 0) {
398  p = line;
399  for (i=0; i < expect_bytes; i++) {
400  if (isxdigit(p[0]) && isxdigit(p[1])) {
401  if (data) *data++ = hex2(p);
402  } else {
403  die("non-hex char on line %d", linenum);
404  }
405  p += 2;
406  }
407  expect_lines--;
408  } else {
409  if (strncmp(line, "ENDCHAR", 7) == 0) {
410  state = 0;
411  } else {
412  die("expected ENDCHAR at line %d", linenum);
413  }
414  }
415  }
416  }
417  if (found_ascent && found_descent) {
418  line_space = font_ascent + font_descent;
419  //printf("// line_space = %d\n", line_space);
420  }
421  if (ENCODING_START <= 'E' && ENCODING_END >= 'E' && g['E'].data != NULL) {
422  cap_height = g['E'].height + g['E'].yoffset;
423  //printf("// cap_height = %d\n", cap_height);
424  }
425  sprintf(name, "_%d", font_size);
426  strcat(font_name, name);
427  if (is_bold) strcat(font_name, "_Bold");
428  if (is_italic) strcat(font_name, "_Italic");
429 }

432 {
433  uint8_t b = 0;
434  if (c >= '0' && c <= '9') {
435  b = c - '0';
436  } else if (c >= 'A' && c <= 'F') {
437  b = c - 'A' + 10;
438  } else if (c >= 'a' && c <= 'f') {
439  b = c - 'a' + 10;
440  }
