134 uint32_t ExFatPartition::bitmapFind(uint32_t cluster, uint32_t count) {
135  uint32_t start = cluster ? cluster - 2 : m_bitmapStart;
136  if (start >= m_clusterCount) {
137  start = 0;
138  }
139  uint32_t endAlloc = start;
140  uint32_t bgnAlloc = start;
141  uint16_t sectorSize = 1 << m_bytesPerSectorShift;
142  size_t i = (start >> 3) & (sectorSize - 1);
143  uint8_t* cache;
144  uint8_t mask = 1 << (start & 7);
145  while (true) {
146  uint32_t sector = m_clusterHeapStartSector +
147  (endAlloc >> (m_bytesPerSectorShift + 3));
148  cache = bitmapCacheGet(sector, FsCache::CACHE_FOR_READ);
149  if (!cache) {
150  return 0;
151  }
152  for (; i < sectorSize; i++) {
153  for (; mask; mask <<= 1) {
154  endAlloc++;
155  if (!(mask & cache[i])) {
156  if ((endAlloc - bgnAlloc) == count) {
157  if (cluster == 0 && count == 1) {
158  // Start at found sector. bitmapModify may increase this.
159  m_bitmapStart = bgnAlloc;
160  }
161  return bgnAlloc + 2;
162  }
163  } else {
164  bgnAlloc = endAlloc;
165  }
166  if (endAlloc == start) {
167  return 1;
168  }
169  if (endAlloc >= m_clusterCount) {
170  endAlloc = bgnAlloc = 0;
171  i = sectorSize;
172  break;
173  }
174  }
175  mask = 1;
176  }
177  i = 0;
178  }
179  return 0;
180 }

182 bool ExFatPartition::bitmapModify(uint32_t cluster,
183  uint32_t count, bool value) {
184  uint32_t sector;
185  uint32_t start = cluster - 2;
186  size_t i;
187  uint8_t* cache;
188  uint8_t mask;
189  cluster -= 2;
190  if ((start + count) > m_clusterCount) {
191  DBG_FAIL_MACRO;
192  goto fail;
193  }
194  if (value) {
195  if (start <= m_bitmapStart && m_bitmapStart < (start + count)) {
196  m_bitmapStart = (start + count) < m_clusterCount ? start + count : 0;
197  }
198  } else {
199  if (start < m_bitmapStart) {
200  m_bitmapStart = start;
201  }
202  }
203  mask = 1 << (start & 7);
204  sector = m_clusterHeapStartSector +
205  (start >> (m_bytesPerSectorShift + 3));
206  i = (start >> 3) & m_sectorMask;
207  while (true) {
208  cache = bitmapCacheGet(sector++, FsCache::CACHE_FOR_WRITE);
209  if (!cache) {
210  DBG_FAIL_MACRO;
211  goto fail;
212  }
213  for (; i < m_bytesPerSector; i++) {
214  for (; mask; mask <<= 1) {
215  if (value == static_cast<bool>(cache[i] & mask)) {
216  DBG_FAIL_MACRO;
217  goto fail;
218  }
219  cache[i] ^= mask;
220  if (--count == 0) {
221  return true;
222  }
223  }
224  mask = 1;
225  }
226  i = 0;
227  }
228 
229  fail:
