334 // send command and return error code. Return zero for OK
335 uint8_t SdSpiCard::cardCommand(uint8_t cmd, uint32_t arg) {
336  // select card
337  if (!m_spiActive) {
338  spiStart();
339  }
340  // wait if busy unless CMD0
341  if (cmd != CMD0) {
342  DBG_BEGIN_TIME(DBG_CMD_BUSY);
343  waitNotBusy(SD_CMD_TIMEOUT);
344  DBG_END_TIME(DBG_CMD_BUSY);
345  }
346 
347 #if USE_SD_CRC
348  // form message
349  uint8_t buf[6];
350  buf[0] = (uint8_t)0x40U | cmd;
351  buf[1] = (uint8_t)(arg >> 24U);
352  buf[2] = (uint8_t)(arg >> 16U);
353  buf[3] = (uint8_t)(arg >> 8U);
354  buf[4] = (uint8_t)arg;
355 
356  // add CRC
357  buf[5] = CRC7(buf, 5);
358 
359  // send message
360  spiSend(buf, 6);
361 #else // USE_SD_CRC
362  // send command
363  spiSend(cmd | 0x40);
364 
365  // send argument
366  uint8_t *pa = reinterpret_cast<uint8_t *>(&arg);
367  for (int8_t i = 3; i >= 0; i--) {
368  spiSend(pa[i]);
369  }
370 
371  // send CRC - correct for CMD0 with arg zero or CMD8 with arg 0X1AA
372  spiSend(cmd == CMD0 ? 0X95 : 0X87);
373 #endif // USE_SD_CRC
374 
375  // discard first fill byte to avoid MISO pull-up problem.
376  spiReceive();
377 
378  // there are 1-8 fill bytes before response. fill bytes should be 0XFF.
379  for (uint8_t i = 0; ((m_status = spiReceive()) & 0X80) && i < 10; i++) {
380  }
381  return m_status;
382 }

637  SdMillis_t t0 = SysCall::curTimeMS();

694 #if ENABLE_DEDICATED_SPI

708  }

712 #else // ENABLE_DEDICATED_SPI

714  return false;
715  }
716  for (size_t i = 0; i < ns; i++, dst += 512) {
717  if (!readData(dst, 512)) {
718  return false;
719  }
720  }
721  return readStop();
722 #endif // ENABLE_DEDICATED_SPI
723 }
