120 // http://forum.arduino.cc/index.php?topic=167414.msg1246851#msg1246851
121 // from robtillaart post:
122 // The code is based upon the divu10() code from the book Hackers Delight1.
123 // My insight was that the error formula in divu10() was in fact modulo 10
124 // but not always. Sometimes it was 10 more.
125 void divmod10(uint32_t in, uint32_t &div, uint32_t &mod)
126 {
127  // q = in * 0.8;
128  uint32_t q = (in >> 1) + (in >> 2);
129  q = q + (q >> 4);
130  q = q + (q >> 8);
131  q = q + (q >> 16); // not needed for 16 bit version
132 
133  // q = q / 8; ==> q = in *0.1;
134  q = q >> 3;
135 
136  // determine error
137  uint32_t r = in - ((q << 3) + (q << 1)); // r = in - q*10;
138  div = q + (r > 9);
139  if (r > 9) mod = r - 10;
140  else mod = r;
141 }
142 // Hackers delight function is here:
143 // http://www.hackersdelight.org/hdcodetxt/divuc.c.txt
144 // Code below uses 8/10 = 0.1100 1100 1100 1100 1100 1100 1100 1100.
145 // 15 ops including the multiply, or 17 elementary ops.
146 unsigned divu10(unsigned n) {
147  unsigned q, r;
148 
149  q = (n >> 1) + (n >> 2);
150  q = q + (q >> 4);
151  q = q + (q >> 8);
152  q = q + (q >> 16);
153  q = q >> 3;
154  r = n - q*10;
155  return q + ((r + 6) >> 4);
156 // return q + (r > 9);
157 }

160 // Format 16-bit unsigned
161 char* fmtBase10(char* str, uint16_t n) {
162  while (n > 9) {
163 #ifdef USE_STIMMER
164  uint8_t tmp8, r;
165  divmod10_asm16(n, r, tmp8);
166 #else // USE_STIMMER
167  uint16_t t = n;
168  n = (n >> 1) + (n >> 2);
169  n = n + (n >> 4);
170  n = n + (n >> 8);
171  // n = n + (n >> 16); // no code for 16-bit n
172  n = n >> 3;
173  uint8_t r = t - (((n << 2) + n) << 1);
174  if (r > 9) {
175  n++;
176  r -= 10;
177  }
178 #endif // USE_STIMMER
179  *--str = r + '0';
180  }
181  *--str = n + '0';
182  return str;
183 }
