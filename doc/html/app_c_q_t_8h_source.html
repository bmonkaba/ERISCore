5 // with a maximum error of 0.1620 degrees

7 {
8  static const uint32_t sign_mask = 0x80000000;
9  static const float b = 0.596227f;
10 
11  // Extract the sign bits
12  uint32_t ux_s = sign_mask & (uint32_t &)x;
13  uint32_t uy_s = sign_mask & (uint32_t &)y;
14 
15  // Determine the quadrant offset
16  float q = (float)( ( ~ux_s & uy_s ) >> 29 | ux_s >> 30 );
17 
18  // Calculate the arctangent in the first quadrant
19  float bxy_a = ::fabs( b * x * y );
20  float num = bxy_a + y * y;
21  float atan_1q = num / ( x * x + bxy_a + num );
22 
23  // Translate it to the proper quadrant
24  uint32_t uatan_2q = (ux_s ^ uy_s) | (uint32_t &)atan_1q;
25  return q + (float &)uatan_2q;
26 }
