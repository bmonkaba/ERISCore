283  // color565toRGB - converts 565 format 16 bit color to RGB
284  static void color565toRGB(uint16_t color, uint8_t &r, uint8_t &g,
285  uint8_t &b) {
286  r = (color >> 8) & 0x00F8;
287  g = (color >> 3) & 0x00FC;
288  b = (color << 3) & 0x00F8;
289  }
290 
291  // color565toRGB14 - converts 16 bit 565 format color to 14 bit RGB (2
292  // bits clear for math and sign)
293  // returns 00rrrrr000000000,00gggggg00000000,00bbbbb000000000
294  // thus not overloading sign, and allowing up to double for additions for
295  // fixed point delta
296  static void color565toRGB14(uint16_t color, int16_t &r, int16_t &g,
297  int16_t &b) {
298  r = (color >> 2) & 0x3E00;
299  g = (color << 3) & 0x3F00;
300  b = (color << 9) & 0x3E00;
301  }
302 
303  // RGB14tocolor565 - converts 14 bit RGB back to 16 bit 565 format
304  // color
305  static uint16_t RGB14tocolor565(int16_t r, int16_t g, int16_t b) {
306  return (((r & 0x3E00) << 2) | ((g & 0x3F00) >> 3) | ((b & 0x3E00) >> 9));
307  }

895  waitTransmitComplete();
896  _pkinetisl_spi->C2 = SPI_C2_SPIMODE; // make sure 8 bit mode.

898  uint8_t s;
899  do {
900  s = _pkinetisl_spi->S;
901  // wait if output buffer busy.
902  // Clear out buffer if there is something there...
903  if ((s & SPI_S_SPRF)) {
904  uint8_t d __attribute__((unused));
905  d = _pkinetisl_spi->DL;
906  d = _pkinetisl_spi->DH;
907  _data_sent_not_completed--; // let system know we sent something

910  } while (!(s & SPI_S_SPTEF) || (s & SPI_S_SPRF));
911 
912  _pkinetisl_spi->DL = data; // output low byte
913  _pkinetisl_spi->DH = data >> 8; // output high byte
914  _data_sent_not_completed++; // let system know we sent something
