37  audio_block_t *block;

40 
41  uint32_t offset;

43 
44  //capture only if channel 0 is connected
45  block = receiveReadOnly(0);
46  if (!block){
47  blockb = receiveReadOnly(1);
48  if (blockb) release(blockb);
49  return;
50  }
51 
52  blockb = receiveReadOnly(1);
53  if (!blockb) {
54  isDualChannel = false;
55  } else isDualChannel = true;
56 
57  offset = 0;
58  while (offset < AUDIO_BLOCK_SAMPLES) {
59  remain = AUDIO_BLOCK_SAMPLES - offset;
60  switch (state) {
61  case STATE_WAIT_TRIGGER:
62  // TODO: implement this....
63  offset = AUDIO_BLOCK_SAMPLES;
64  break;
65 
66  case STATE_DELAY:
67  //Serial.printf("STATE_DELAY, count = %u\n", count);
68  if (remain < count) {
69  count -= remain;
70  offset = AUDIO_BLOCK_SAMPLES;
71  } else {
72  offset += count;
73  count = mem_length;
74  state = STATE_PRINTING;
75  }
76  break;
77 
78  case STATE_PRINTING:
79  //for the first sample try to find a zero crossing offset in the curernt audio block
80  if (count == mem_length){
81  bool found = false;
82  for(int16_t i=1;i < AUDIO_BLOCK_SAMPLES-h_div;i++){
83  if (block->data[i] > 0 && block->data[i-1] < 0){
84  offset = i;
85  found = true;
86  break;
87  }
88  }
89  if(!found) {offset = AUDIO_BLOCK_SAMPLES;};
90  edgeCount = 0;
91  peakValue = 0;
92  }
93 
94  while ((offset < AUDIO_BLOCK_SAMPLES) && (count > 0)){
95  h_div_count++;
96  if(h_div_count==h_div){
97  h_div_count=0;
98  count--;
99  memory[0][mem_length - count - 1 ] = block->data[offset];
100  peakValue = max(peakValue,abs(block->data[offset]));
101  if (isDualChannel){
102  memory[1][mem_length - count - 1 ] = blockb->data[offset];
103  peakValue = max(peakValue,abs(blockb->data[offset]));
104  }else memory[1][mem_length - count - 1 ] = 0;
105 
106  if ((offset >= h_div) && (block->data[offset] >= 0 ) && (block->data[offset-h_div] < 0)) edgeCount++;
107  }
108  offset++;
109  }
110 
111 
112  if (count == 0){
113  //Serial.println(edgeCount);
114  if((edgeCount < 3) && (auto_h_div < 4)) auto_h_div++;
115  if((edgeCount > 4) && (auto_h_div > 2)) auto_h_div--;
116  isAvailable = true;
117  if (autoTrigger){
118  trigger();
119  }else state = STATE_IDLE;
120  }
121  break;
122 
123  default: // STATE_IDLE
124  offset = AUDIO_BLOCK_SAMPLES;
125  break;
126  }
127  }
128  release(block);
129  if(isDualChannel) release(blockb);
130 }
