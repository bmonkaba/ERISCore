125  float read(unsigned int binFirst, unsigned int binLast,FFTReadRange *fftRR = NULL) {
126  if (binFirst > binLast) {
127  unsigned int tmp = binLast;
128  binLast = binFirst;
129  binFirst = tmp;
130  }
131  if(fftRR){
132  fftRR->peakBin = 0;
133  fftRR->peakFrequency = 0;
134  fftRR->peakValue = 0;
135  }
136  if (binFirst > 511) return 0.0;
137  if (binLast > 511) binLast = 511;

139  float comp;
140  do {
141  comp = (float)output[binFirst];
142  maxf = max(comp,maxf);
143  if(fftRR) {
144  if (maxf > fftRR->peakValue) {
145  fftRR->peakBin = binFirst;
146  fftRR->peakFFTResult = output_packed[binFirst];
147  fftRR->peakValue = maxf;
148  }
149  }
150  binFirst++;
151  } while (binFirst <= binLast);
152  if(fftRR) fftRR->peakValue *= (1.0 / 16384.0);
153  return maxf * (1.0 / 16384.0);

159  float read(float freq_from, float freq_to, FFTReadRange *fftRR = NULL) {
160  //convert f to bin
161  float bw;
162  float bin_size;
163  unsigned int start_bin;
164  unsigned int stop_bin;
165 
166  bw = (AUDIO_SAMPLE_RATE_EXACT * 0.5) / (float)subsample_by;
167  bin_size = bw/512;
168  start_bin = (unsigned int)(freq_from / bin_size);
169  stop_bin = (unsigned int)(freq_to / bin_size);
170 
171  if(fftRR){
172  fftRR->startBin = start_bin;
173  fftRR->stopBin = stop_bin;
174  fftRR->startFrequency = freq_from;
175  fftRR->stopFrequency = freq_to;
176  }
177 
178 
179  if (freq_to > bw){
180  if(fftRR){
181  fftRR->startBin = 0;
182  fftRR->stopBin = 0;
183  fftRR->startFrequency = 0;
184  fftRR->stopFrequency = 0;
185  fftRR->peakFrequency = 0;
186  fftRR->estimatedFrequency = 0;
187  fftRR->peakValue = 0;
188  fftRR->avgValueSlow = 0;
189  fftRR->avgValueFast = 0;
190  }
191  return 0;
192  }

194  /*
195  Serial.print(F("fft read: "));
196  Serial.print(subsample_by);Serial.print(F(","));
197  Serial.print(bw);Serial.print(F(","));
198  Serial.print(bin_size);Serial.print(F(","));
199  Serial.print(start_bin);Serial.print(F(","));
200  Serial.println(stop_bin);
201  */
202  float rval = read(start_bin,stop_bin,fftRR);
203  if(fftRR){
204  //from the peak bin calc the freq
205  fftRR->peakFrequency = (fftRR->peakBin * bin_size) - (bin_size/2.0); //center of the bin
206  if (fftRR->peakFrequency < 0) fftRR->peakFrequency = 0.01;
207  if ((fftRR->peakBin > 1) && (fftRR->peakBin < 510) && (output[fftRR->peakBin] > 0)){
208  //from the balance of the side lobes, estimate the actual frequency
209  float ratio = 1;
210  float lobeFrequency = 1;
211  if ((output[fftRR->peakBin+1]-output[fftRR->peakBin-1]) > 0.1){
212  //pos lobe
213  ratio = (output[fftRR->peakBin+1] - output[fftRR->peakBin-1]) / (1.0 * output[fftRR->peakBin]);
214  lobeFrequency = ((fftRR->peakBin+1) * bin_size) - bin_size/2;
215  } else if (output[fftRR->peakBin-1] > 0.1){
216  //neg lobe
217  ratio = (output[fftRR->peakBin-1]-output[fftRR->peakBin+1]) / (1.0 * output[fftRR->peakBin]);
218  lobeFrequency = ((fftRR->peakBin-1) * bin_size) - bin_size/2;
219  }
220  fftRR->estimatedFrequency = (fftRR->peakFrequency * (1-ratio)) + (lobeFrequency * ratio);
221  //clamp estimate
222  if (fftRR->estimatedFrequency > fftRR->stopFrequency)fftRR->estimatedFrequency = fftRR->stopFrequency;
223  if (fftRR->estimatedFrequency < fftRR->startFrequency)fftRR->estimatedFrequency = fftRR->startFrequency;
224 
225  fftRR->avgValueFast = (fftRR->avgValueFast * 0.80) + (fftRR->peakValue * 0.20); //used to calc moving average convergence / divergence (MACD)
226  fftRR->avgValueSlow = (fftRR->avgValueSlow * 0.95) + (fftRR->peakValue * 0.05); //by comparing a short and long moving average; slow transient detection
227  if(fftRR->peakValue > fftRR->avgValueFast) fftRR->avgValueFast = (fftRR->avgValueFast * 0.50) + (fftRR->peakValue * 0.50);
228  if(fftRR->peakValue > fftRR->avgValueSlow) fftRR->avgValueSlow = (fftRR->avgValueSlow * 0.7) + (fftRR->peakValue * 0.3);
229 
230  fftRR->macdValue = fftRR->avgValueFast - fftRR->avgValueSlow;
231  fftRR->transientValue = fftRR->peakValue - fftRR->avgValueSlow;
232  }
233  }
234 
235  return rval;
236  }
