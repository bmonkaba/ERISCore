455 inline uint32_t sdCardCapacity(csd_t* csd) {
456  if (csd->v1.csd_ver == 0) {
457  uint8_t read_bl_len = csd->v1.read_bl_len;
458  uint16_t c_size = (csd->v1.c_size_high << 10)
459  | (csd->v1.c_size_mid << 2) | csd->v1.c_size_low;
460  uint8_t c_size_mult = (csd->v1.c_size_mult_high << 1)
461  | csd->v1.c_size_mult_low;
462  return (uint32_t)(c_size + 1) << (c_size_mult + read_bl_len - 7);
463  } else if (csd->v2.csd_ver == 1) {
464  return (((uint32_t)csd->v2.c_size_high << 16) +
465  ((uint16_t)csd->v2.c_size_mid << 8) + csd->v2.c_size_low + 1) << 10;
466  } else {
467  return 0;
468  }
