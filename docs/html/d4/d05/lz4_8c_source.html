63 /* LZ4_FORCE_MEMORY_ACCESS
64  * By default, access to unaligned memory is controlled by `memcpy()`, which is safe and portable.
65  * Unfortunately, on some target/compiler combinations, the generated assembly is sub-optimal.
66  * The below switch allow to select different access method for improved performance.
67  * Method 0 (default) : use `memcpy()`. Safe and portable.
68  * Method 1 : `__packed` statement. It depends on compiler extension (ie, not portable).
69  * This method is safe if your compiler supports it, and *generally* as fast or faster than `memcpy`.
70  * Method 2 : direct access. This method is portable but violate C standard.
71  * It can generate buggy code on targets which assembly generation depends on alignment.
72  * But in some circumstances, it's the only known way to get the most performance (ie GCC + ARMv6)
73  * See https://fastcompression.blogspot.fr/2015/08/accessing-unaligned-memory.html for details.
74  * Prefer these methods in priority order (0 > 1 > 2)
75  */
76 #ifndef LZ4_FORCE_MEMORY_ACCESS /* can be defined externally */
77 # if defined(__GNUC__) && \
78  ( defined(__ARM_ARCH_6__) || defined(__ARM_ARCH_6J__) || defined(__ARM_ARCH_6K__) \
79  || defined(__ARM_ARCH_6Z__) || defined(__ARM_ARCH_6ZK__) || defined(__ARM_ARCH_6T2__) )

145 /* LZ4_FORCE_O2 and LZ4_FORCE_INLINE
146  * gcc on ppc64le generates an unrolled SIMDized loop for LZ4_wildCopy8,
147  * together with a simple 8-byte copy loop as a fall-back path.
148  * However, this optimization hurts the decompression speed by >30%,
149  * because the execution does not go to the optimized loop
150  * for typical compressible data, and all of the preamble checks

439 LZ4_memcpy_using_offset_base(BYTE* dstPtr, const BYTE* srcPtr, BYTE* dstEnd, const size_t offset)
440 {

474 /* LZ4_memcpy_using_offset() presumes :
475  * - dstEnd >= dstPtr + MINMATCH
476  * - there is at least 8 bytes available to write after dstEnd */
477 LZ4_FORCE_INLINE void
478 LZ4_memcpy_using_offset(BYTE* dstPtr, const BYTE* srcPtr, BYTE* dstEnd, const size_t offset)
479 {
480  BYTE v[8];
481 
482  assert(dstEnd >= dstPtr + MINMATCH);
483 
484  switch(offset) {
485  case 1:
486  MEM_INIT(v, *srcPtr, 8);
487  break;
488  case 2:
489  LZ4_memcpy(v, srcPtr, 2);
490  LZ4_memcpy(&v[2], srcPtr, 2);
491  LZ4_memcpy(&v[4], v, 4);
492  break;
493  case 4:
494  LZ4_memcpy(v, srcPtr, 4);
495  LZ4_memcpy(&v[4], srcPtr, 4);
496  break;
497  default:
498  LZ4_memcpy_using_offset_base(dstPtr, srcPtr, dstEnd, offset);
499  return;
500  }
501 
502  LZ4_memcpy(dstPtr, v, 8);
503  dstPtr += 8;
504  while (dstPtr < dstEnd) {
505  LZ4_memcpy(dstPtr, v, 8);
506  dstPtr += 8;
507  }
508 }

518  if (LZ4_isLittleEndian()) {

521 # if defined(__clang__) && (__clang_major__ < 10)
522  /* Avoid undefined clang-cl intrinics issue.
523  * See https://github.com/lz4/lz4/pull/1017 for details. */
524  return (unsigned)__builtin_ia32_tzcnt_u64(val) >> 3;
525 # else

527  return (unsigned)_tzcnt_u64(val) >> 3;

533 # elif (defined(__clang__) || (defined(__GNUC__) && ((__GNUC__ > 3) || \
534  ((__GNUC__ == 3) && (__GNUC_MINOR__ >= 4))))) && \
535  !defined(LZ4_FORCE_SW_BITCOUNT)
536  return (unsigned)__builtin_ctzll((U64)val) >> 3;
537 # else
538  const U64 m = 0x0101010101010101ULL;
539  val ^= val - 1;
540  return (unsigned)(((U64)((val & (m - 1)) * m)) >> 56);
541 # endif
542  } else /* 32 bits */ {
543 # if defined(_MSC_VER) && (_MSC_VER >= 1400) && !defined(LZ4_FORCE_SW_BITCOUNT)
544  unsigned long r;
545  _BitScanForward(&r, (U32)val);

555  }

575  };

577  U64 const t = (((val >> 8) - mask) | val) & mask;
578  return ctz7_tab[(t * 0x0080402010080402ULL) >> 57];
579 #else
580  /* this method doesn't consume memory space like the previous one,
581  * but it contains several branches,
582  * that may end up slowing execution */
583  static const U32 by32 = sizeof(val)*4; /* 32 on 64 bits (goal), 16 on 32 bits.
584  Just to avoid some static analyzer complaining about shift by 32 on 32-bits target.
585  Note that this code path is never triggered in 32-bits mode. */
586  unsigned r;
587  if (!(val>>by32)) { r=4; } else { r=0; val>>=by32; }
588  if (!(val>>16)) { r+=2; val>>=8; } else { val>>=24; }
589  r += (!val);
590  return r;
591 #endif
592 # endif
593  } else /* 32 bits */ {
594 # if (defined(__clang__) || (defined(__GNUC__) && ((__GNUC__ > 3) || \
595  ((__GNUC__ == 3) && (__GNUC_MINOR__ >= 4))))) && \
596  !defined(LZ4_FORCE_SW_BITCOUNT)
597  return (unsigned)__builtin_clz((U32)val) >> 3;
598 # else
599  val >>= 8;
600  val = ((((val + 0x00FFFF00) | 0x00FFFFFF) + val) |
601  (val + 0x00FF0000)) >> 24;
602  return (unsigned)val ^ 3;
603 # endif
604  }
605  }

609 #define STEPSIZE sizeof(reg_t)
610 LZ4_FORCE_INLINE
611 unsigned LZ4_count(const BYTE* pIn, const BYTE* pMatch, const BYTE* pInLimit)
612 {
613  const BYTE* const pStart = pIn;
614 
615  if (likely(pIn < pInLimit-(STEPSIZE-1))) {
616  reg_t const diff = LZ4_read_ARCH(pMatch) ^ LZ4_read_ARCH(pIn);
617  if (!diff) {
618  pIn+=STEPSIZE; pMatch+=STEPSIZE;
619  } else {
620  return LZ4_NbCommonBytes(diff);
621  } }
622 
623  while (likely(pIn < pInLimit-(STEPSIZE-1))) {
624  reg_t const diff = LZ4_read_ARCH(pMatch) ^ LZ4_read_ARCH(pIn);
625  if (!diff) { pIn+=STEPSIZE; pMatch+=STEPSIZE; continue; }
626  pIn += LZ4_NbCommonBytes(diff);
627  return (unsigned)(pIn - pStart);
628  }
629 
630  if ((STEPSIZE==8) && (pIn<(pInLimit-3)) && (LZ4_read32(pMatch) == LZ4_read32(pIn))) { pIn+=4; pMatch+=4; }
631  if ((pIn<(pInLimit-1)) && (LZ4_read16(pMatch) == LZ4_read16(pIn))) { pIn+=2; pMatch+=2; }
632  if ((pIn<pInLimit) && (*pMatch == *pIn)) pIn++;
633  return (unsigned)(pIn - pStart);
634 }

813 LZ4_FORCE_INLINE void
814 LZ4_prepareTable(LZ4_stream_t_internal* const cctx,
815  const int inputSize,
816  const tableType_t tableType) {
817  /* If the table hasn't been used, it's guaranteed to be zeroed out, and is
818  * therefore safe to use no matter what mode we're in. Otherwise, we figure
819  * out if it's safe to leave as is or whether it needs to be reset.
820  */
821  if ((tableType_t)cctx->tableType != clearedTable) {
822  assert(inputSize >= 0);
823  if ((tableType_t)cctx->tableType != tableType
824  || ((tableType == byU16) && cctx->currentOffset + (unsigned)inputSize >= 0xFFFFU)
825  || ((tableType == byU32) && cctx->currentOffset > 1 GB)
826  || tableType == byPtr
827  || inputSize >= 4 KB)
828  {
829  DEBUGLOG(4, "LZ4_prepareTable: Resetting table in %p", cctx);
830  MEM_INIT(cctx->hashTable, 0, LZ4_HASHTABLESIZE);
831  cctx->currentOffset = 0;
832  cctx->tableType = (U32)clearedTable;
833  } else {
834  DEBUGLOG(4, "LZ4_prepareTable: Re-use hash table (no reset)");
835  }
836  }
837 
838  /* Adding a gap, so all previous entries are > LZ4_DISTANCE_MAX back,
839  * is faster than compressing without a gap.
840  * However, compressing with currentOffset == 0 is faster still,
841  * so we preserve that case.
842  */
843  if (cctx->currentOffset != 0 && tableType == byU32) {
844  DEBUGLOG(5, "LZ4_prepareTable: adding 64KB to currentOffset");
845  cctx->currentOffset += 64 KB;
846  }
847 
848  /* Finally, clear history */
849  cctx->dictCtx = NULL;
850  cctx->dictionary = NULL;
851  cctx->dictSize = 0;
852 }

860 LZ4_FORCE_INLINE int LZ4_compress_generic_validated(
861  LZ4_stream_t_internal* const cctx,
862  const char* const source,
863  char* const dest,
864  const int inputSize,
865  int *inputConsumed, /* only written when outputDirective == fillOutput */
866  const int maxOutputSize,
867  const limitedOutput_directive outputDirective,
868  const tableType_t tableType,
869  const dict_directive dictDirective,
870  const dictIssue_directive dictIssue,
871  const int acceleration)
872 {
873  int result;
874  const BYTE* ip = (const BYTE*) source;
875 
876  U32 const startIndex = cctx->currentOffset;
877  const BYTE* base = (const BYTE*) source - startIndex;
878  const BYTE* lowLimit;
879 
880  const LZ4_stream_t_internal* dictCtx = (const LZ4_stream_t_internal*) cctx->dictCtx;
881  const BYTE* const dictionary =
882  dictDirective == usingDictCtx ? dictCtx->dictionary : cctx->dictionary;
883  const U32 dictSize =
884  dictDirective == usingDictCtx ? dictCtx->dictSize : cctx->dictSize;
885  const U32 dictDelta = (dictDirective == usingDictCtx) ? startIndex - dictCtx->currentOffset : 0; /* make indexes in dictCtx comparable with index in current context */
886 
887  int const maybe_extMem = (dictDirective == usingExtDict) || (dictDirective == usingDictCtx);
888  U32 const prefixIdxLimit = startIndex - dictSize; /* used when dictDirective == dictSmall */
889  const BYTE* const dictEnd = dictionary ? dictionary + dictSize : dictionary;
890  const BYTE* anchor = (const BYTE*) source;
891  const BYTE* const iend = ip + inputSize;
892  const BYTE* const mflimitPlusOne = iend - MFLIMIT + 1;
893  const BYTE* const matchlimit = iend - LASTLITERALS;
894 
895  /* the dictCtx currentOffset is indexed on the start of the dictionary,
896  * while a dictionary in the current context precedes the currentOffset */
897  const BYTE* dictBase = (dictionary == NULL) ? NULL :
898  (dictDirective == usingDictCtx) ?
899  dictionary + dictSize - dictCtx->currentOffset :
900  dictionary + dictSize - startIndex;
901 
902  BYTE* op = (BYTE*) dest;
903  BYTE* const olimit = op + maxOutputSize;
904 
905  U32 offset = 0;
906  U32 forwardH;
907 
908  DEBUGLOG(5, "LZ4_compress_generic_validated: srcSize=%i, tableType=%u", inputSize, tableType);
909  assert(ip != NULL);
910  /* If init conditions are not met, we don't have to mark stream
911  * as having dirty context, since no action was taken yet */
912  if (outputDirective == fillOutput && maxOutputSize < 1) { return 0; } /* Impossible to store anything */
913  if ((tableType == byU16) && (inputSize>=LZ4_64Klimit)) { return 0; } /* Size too large (not within 64K limit) */
914  if (tableType==byPtr) assert(dictDirective==noDict); /* only supported use case with byPtr */
915  assert(acceleration >= 1);
916 
917  lowLimit = (const BYTE*)source - (dictDirective == withPrefix64k ? dictSize : 0);
918 
919  /* Update context state */
920  if (dictDirective == usingDictCtx) {
921  /* Subsequent linked blocks can't use the dictionary. */
922  /* Instead, they use the block we just compressed. */
923  cctx->dictCtx = NULL;
924  cctx->dictSize = (U32)inputSize;
925  } else {
926  cctx->dictSize += (U32)inputSize;
927  }
928  cctx->currentOffset += (U32)inputSize;
929  cctx->tableType = (U32)tableType;
930 
931  if (inputSize<LZ4_minLength) goto _last_literals; /* Input too small, no compression (all literals) */
932 
933  /* First Byte */
934  LZ4_putPosition(ip, cctx->hashTable, tableType, base);
935  ip++; forwardH = LZ4_hashPosition(ip, tableType);
936 
937  /* Main Loop */
938  for ( ; ; ) {
939  const BYTE* match;
940  BYTE* token;
941  const BYTE* filledIp;
942 
943  /* Find a match */
944  if (tableType == byPtr) {
945  const BYTE* forwardIp = ip;
946  int step = 1;
947  int searchMatchNb = acceleration << LZ4_skipTrigger;
948  do {
949  U32 const h = forwardH;
950  ip = forwardIp;
951  forwardIp += step;
952  step = (searchMatchNb++ >> LZ4_skipTrigger);
953 
954  if (unlikely(forwardIp > mflimitPlusOne)) goto _last_literals;
955  assert(ip < mflimitPlusOne);
956 
957  match = LZ4_getPositionOnHash(h, cctx->hashTable, tableType, base);
958  forwardH = LZ4_hashPosition(forwardIp, tableType);
959  LZ4_putPositionOnHash(ip, h, cctx->hashTable, tableType, base);
960 
961  } while ( (match+LZ4_DISTANCE_MAX < ip)
962  || (LZ4_read32(match) != LZ4_read32(ip)) );
963 
964  } else { /* byU32, byU16 */
965 
966  const BYTE* forwardIp = ip;
967  int step = 1;
968  int searchMatchNb = acceleration << LZ4_skipTrigger;
969  do {
970  U32 const h = forwardH;
971  U32 const current = (U32)(forwardIp - base);
972  U32 matchIndex = LZ4_getIndexOnHash(h, cctx->hashTable, tableType);
973  assert(matchIndex <= current);
974  assert(forwardIp - base < (ptrdiff_t)(2 GB - 1));
975  ip = forwardIp;
976  forwardIp += step;
977  step = (searchMatchNb++ >> LZ4_skipTrigger);
978 
979  if (unlikely(forwardIp > mflimitPlusOne)) goto _last_literals;
980  assert(ip < mflimitPlusOne);
981 
982  if (dictDirective == usingDictCtx) {
983  if (matchIndex < startIndex) {
984  /* there was no match, try the dictionary */
985  assert(tableType == byU32);
986  matchIndex = LZ4_getIndexOnHash(h, dictCtx->hashTable, byU32);
987  match = dictBase + matchIndex;
988  matchIndex += dictDelta; /* make dictCtx index comparable with current context */
989  lowLimit = dictionary;
990  } else {
991  match = base + matchIndex;
992  lowLimit = (const BYTE*)source;
993  }
994  } else if (dictDirective == usingExtDict) {
995  if (matchIndex < startIndex) {
996  DEBUGLOG(7, "extDict candidate: matchIndex=%5u < startIndex=%5u", matchIndex, startIndex);
997  assert(startIndex - matchIndex >= MINMATCH);
998  assert(dictBase);
999  match = dictBase + matchIndex;
1000  lowLimit = dictionary;
1001  } else {
1002  match = base + matchIndex;
1003  lowLimit = (const BYTE*)source;
1004  }
1005  } else { /* single continuous memory segment */
1006  match = base + matchIndex;
1007  }
1008  forwardH = LZ4_hashPosition(forwardIp, tableType);
1009  LZ4_putIndexOnHash(current, h, cctx->hashTable, tableType);
1010 
1011  DEBUGLOG(7, "candidate at pos=%u (offset=%u \n", matchIndex, current - matchIndex);
1012  if ((dictIssue == dictSmall) && (matchIndex < prefixIdxLimit)) { continue; } /* match outside of valid area */
1013  assert(matchIndex < current);
1014  if ( ((tableType != byU16) || (LZ4_DISTANCE_MAX < LZ4_DISTANCE_ABSOLUTE_MAX))
1015  && (matchIndex+LZ4_DISTANCE_MAX < current)) {
1016  continue;
1017  } /* too far */
1018  assert((current - matchIndex) <= LZ4_DISTANCE_MAX); /* match now expected within distance */
1019 
1020  if (LZ4_read32(match) == LZ4_read32(ip)) {
1021  if (maybe_extMem) offset = current - matchIndex;
1022  break; /* match found */
1023  }
1024 
1025  } while(1);
1026  }
1027 
1028  /* Catch up */
1029  filledIp = ip;
1030  while (((ip>anchor) & (match > lowLimit)) && (unlikely(ip[-1]==match[-1]))) { ip--; match--; }
1031 
1032  /* Encode Literals */
1033  { unsigned const litLength = (unsigned)(ip - anchor);
1034  token = op++;
1035  if ((outputDirective == limitedOutput) && /* Check output buffer overflow */
1036  (unlikely(op + litLength + (2 + 1 + LASTLITERALS) + (litLength/255) > olimit)) ) {
1037  return 0; /* cannot compress within `dst` budget. Stored indexes in hash table are nonetheless fine */
1038  }
1039  if ((outputDirective == fillOutput) &&
1040  (unlikely(op + (litLength+240)/255 /* litlen */ + litLength /* literals */ + 2 /* offset */ + 1 /* token */ + MFLIMIT - MINMATCH /* min last literals so last match is <= end - MFLIMIT */ > olimit))) {
1041  op--;
1042  goto _last_literals;
1043  }
1044  if (litLength >= RUN_MASK) {
1045  int len = (int)(litLength - RUN_MASK);
1046  *token = (RUN_MASK<<ML_BITS);
1047  for(; len >= 255 ; len-=255) *op++ = 255;
1048  *op++ = (BYTE)len;
1049  }
1050  else *token = (BYTE)(litLength<<ML_BITS);
1051 
1052  /* Copy Literals */
1053  LZ4_wildCopy8(op, anchor, op+litLength);
1054  op+=litLength;
1055  DEBUGLOG(6, "seq.start:%i, literals=%u, match.start:%i",
1056  (int)(anchor-(const BYTE*)source), litLength, (int)(ip-(const BYTE*)source));
1057  }
1058 
1059 _next_match:
1060  /* at this stage, the following variables must be correctly set :
1061  * - ip : at start of LZ operation
1062  * - match : at start of previous pattern occurrence; can be within current prefix, or within extDict
1063  * - offset : if maybe_ext_memSegment==1 (constant)
1064  * - lowLimit : must be == dictionary to mean "match is within extDict"; must be == source otherwise
1065  * - token and *token : position to write 4-bits for match length; higher 4-bits for literal length supposed already written
1066  */
1067 
1068  if ((outputDirective == fillOutput) &&
1069  (op + 2 /* offset */ + 1 /* token */ + MFLIMIT - MINMATCH /* min last literals so last match is <= end - MFLIMIT */ > olimit)) {
1070  /* the match was too close to the end, rewind and go to last literals */
1071  op = token;
1072  goto _last_literals;
1073  }
1074 
1075  /* Encode Offset */
1076  if (maybe_extMem) { /* static test */
1077  DEBUGLOG(6, " with offset=%u (ext if > %i)", offset, (int)(ip - (const BYTE*)source));
1078  assert(offset <= LZ4_DISTANCE_MAX && offset > 0);
1079  LZ4_writeLE16(op, (U16)offset); op+=2;
1080  } else {
1081  DEBUGLOG(6, " with offset=%u (same segment)", (U32)(ip - match));
1082  assert(ip-match <= LZ4_DISTANCE_MAX);
1083  LZ4_writeLE16(op, (U16)(ip - match)); op+=2;
1084  }
1085 
1086  /* Encode MatchLength */
1087  { unsigned matchCode;
1088 
1089  if ( (dictDirective==usingExtDict || dictDirective==usingDictCtx)
1090  && (lowLimit==dictionary) /* match within extDict */ ) {
1091  const BYTE* limit = ip + (dictEnd-match);
1092  assert(dictEnd > match);
1093  if (limit > matchlimit) limit = matchlimit;
1094  matchCode = LZ4_count(ip+MINMATCH, match+MINMATCH, limit);
1095  ip += (size_t)matchCode + MINMATCH;
1096  if (ip==limit) {
1097  unsigned const more = LZ4_count(limit, (const BYTE*)source, matchlimit);
1098  matchCode += more;
1099  ip += more;
1100  }
1101  DEBUGLOG(6, " with matchLength=%u starting in extDict", matchCode+MINMATCH);
1102  } else {
1103  matchCode = LZ4_count(ip+MINMATCH, match+MINMATCH, matchlimit);
1104  ip += (size_t)matchCode + MINMATCH;
1105  DEBUGLOG(6, " with matchLength=%u", matchCode+MINMATCH);
1106  }
1107 
1108  if ((outputDirective) && /* Check output buffer overflow */
1109  (unlikely(op + (1 + LASTLITERALS) + (matchCode+240)/255 > olimit)) ) {
1110  if (outputDirective == fillOutput) {
1111  /* Match description too long : reduce it */
1112  U32 newMatchCode = 15 /* in token */ - 1 /* to avoid needing a zero byte */ + ((U32)(olimit - op) - 1 - LASTLITERALS) * 255;
1113  ip -= matchCode - newMatchCode;
1114  assert(newMatchCode < matchCode);
1115  matchCode = newMatchCode;
1116  if (unlikely(ip <= filledIp)) {
1117  /* We have already filled up to filledIp so if ip ends up less than filledIp
1118  * we have positions in the hash table beyond the current position. This is
1119  * a problem if we reuse the hash table. So we have to remove these positions
1120  * from the hash table.
1121  */
1122  const BYTE* ptr;
1123  DEBUGLOG(5, "Clearing %u positions", (U32)(filledIp - ip));
1124  for (ptr = ip; ptr <= filledIp; ++ptr) {

1578 int LZ4_compress_fast_continue (LZ4_stream_t* LZ4_stream,
1579  const char* source, char* dest,
1580  int inputSize, int maxOutputSize,
1581  int acceleration)
1582 {
1583  const tableType_t tableType = byU32;
1584  LZ4_stream_t_internal* const streamPtr = &LZ4_stream->internal_donotuse;
1585  const char* dictEnd = streamPtr->dictSize ? (const char*)streamPtr->dictionary + streamPtr->dictSize : NULL;
1586 
1587  DEBUGLOG(5, "LZ4_compress_fast_continue (inputSize=%i, dictSize=%u)", inputSize, streamPtr->dictSize);
1588 
1589  LZ4_renormDictT(streamPtr, inputSize); /* fix index overflow */
1590  if (acceleration < 1) acceleration = LZ4_ACCELERATION_DEFAULT;
1591  if (acceleration > LZ4_ACCELERATION_MAX) acceleration = LZ4_ACCELERATION_MAX;
1592 
1593  /* invalidate tiny dictionaries */
1594  if ( (streamPtr->dictSize < 4) /* tiny dictionary : not enough for a hash */
1595  && (dictEnd != source) /* prefix mode */
1596  && (inputSize > 0) /* tolerance : don't lose history, in case next invocation would use prefix mode */
1597  && (streamPtr->dictCtx == NULL) /* usingDictCtx */
1598  ) {
1599  DEBUGLOG(5, "LZ4_compress_fast_continue: dictSize(%u) at addr:%p is too small", streamPtr->dictSize, streamPtr->dictionary);
1600  /* remove dictionary existence from history, to employ faster prefix mode */
1601  streamPtr->dictSize = 0;
1602  streamPtr->dictionary = (const BYTE*)source;
1603  dictEnd = source;
1604  }
1605 
1606  /* Check overlapping input/dictionary space */
1607  { const char* const sourceEnd = source + inputSize;
1608  if ((sourceEnd > (const char*)streamPtr->dictionary) && (sourceEnd < dictEnd)) {
1609  streamPtr->dictSize = (U32)(dictEnd - sourceEnd);
1610  if (streamPtr->dictSize > 64 KB) streamPtr->dictSize = 64 KB;
1611  if (streamPtr->dictSize < 4) streamPtr->dictSize = 0;
1612  streamPtr->dictionary = (const BYTE*)dictEnd - streamPtr->dictSize;
1613  }
1614  }
1615 
1616  /* prefix mode : source data follows dictionary */
1617  if (dictEnd == source) {
1618  if ((streamPtr->dictSize < 64 KB) && (streamPtr->dictSize < streamPtr->currentOffset))
1619  return LZ4_compress_generic(streamPtr, source, dest, inputSize, NULL, maxOutputSize, limitedOutput, tableType, withPrefix64k, dictSmall, acceleration);
1620  else
1621  return LZ4_compress_generic(streamPtr, source, dest, inputSize, NULL, maxOutputSize, limitedOutput, tableType, withPrefix64k, noDictIssue, acceleration);
1622  }
1623 
1624  /* external dictionary mode */
1625  { int result;
1626  if (streamPtr->dictCtx) {
1627  /* We depend here on the fact that dictCtx'es (produced by
1628  * LZ4_loadDict) guarantee that their tables contain no references
1629  * to offsets between dictCtx->currentOffset - 64 KB and
1630  * dictCtx->currentOffset - dictCtx->dictSize. This makes it safe
1631  * to use noDictIssue even when the dict isn't a full 64 KB.
1632  */
1633  if (inputSize > 4 KB) {
1634  /* For compressing large blobs, it is faster to pay the setup
