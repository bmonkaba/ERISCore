72 void PXP_isr(){
73  if((PXP_STAT & PXP_STAT_LUT_DMA_LOAD_DONE_IRQ) != 0){
74  PXP_STAT_CLR = PXP_STAT_LUT_DMA_LOAD_DONE_IRQ;
75  }
76  if((PXP_STAT & PXP_STAT_NEXT_IRQ) != 0){
77  PXP_STAT_CLR = PXP_STAT_NEXT_IRQ;
78  }
79  if((PXP_STAT & PXP_STAT_AXI_READ_ERROR) != 0){
80  PXP_STAT_CLR = PXP_STAT_AXI_READ_ERROR;
81  }
82  if((PXP_STAT & PXP_STAT_AXI_WRITE_ERROR) != 0){
83  PXP_STAT_CLR = PXP_STAT_AXI_WRITE_ERROR;
84  }
85  if((PXP_STAT & PXP_STAT_IRQ) != 0){
86  PXP_STAT_CLR = PXP_STAT_IRQ;
87  PXP_done = true;
88  }
89 #if defined(__IMXRT1062__) // Teensy 4.x
90  asm("DSB");
91 #endif
92 }

94 void PXP_init(){

96

104 // PXP_CTRL_SET = PXP_CTRL_IRQ_ENABLE | PXP_CTRL_NEXT_IRQ_ENABLE;

292 void PXP_process(){
293  while ((PXP_NEXT & PXP_NEXT_ENABLED) != 0){}
294  PXP_done = false;
295 
296  if ((uint32_t)next_pxp.OUT_BUF >= 0x20200000u) arm_dcache_flush_delete((void*)next_pxp.OUT_BUF, OUT_BUF_width * OUT_BUF_height * OUT_BUF_bytesPerPixel);
297  if ((uint32_t)next_pxp.OUT_BUF2 >= 0x20200000u) arm_dcache_flush_delete((void*)next_pxp.OUT_BUF2, OUT_BUF2_width * OUT_BUF2_height * OUT_BUF2_bytesPerPixel);
298  if ((uint32_t)next_pxp.PS_BUF >= 0x20200000u) arm_dcache_flush_delete((void*)next_pxp.PS_BUF, PS_BUF_width * PS_BUF_height * PS_BUF_bytesPerPixel);
299  if ((uint32_t)next_pxp.PS_UBUF >= 0x20200000u) arm_dcache_flush_delete((void*)next_pxp.PS_UBUF, PS_UBUF_width * PS_UBUF_height * PS_UBUF_bytesPerPixel);
300  if ((uint32_t)next_pxp.PS_VBUF >= 0x20200000u) arm_dcache_flush_delete((void*)next_pxp.PS_VBUF, PS_VBUF_width * PS_VBUF_height * PS_VBUF_bytesPerPixel);
301  if ((uint32_t)next_pxp.AS_BUF >= 0x20200000u) arm_dcache_flush_delete((void*)next_pxp.AS_BUF, AS_BUF_width * AS_BUF_height * AS_BUF_bytesPerPixel);
302 
303 
304  PXP_NEXT = (uint32_t)&next_pxp;
305 }
