508  bool isWrenResultOK(WrenInterpretResult res){
509  switch (res){
510  case WREN_RESULT_COMPILE_ERROR:
511  if(sci->requestStartLZ4Message()){
512  sci->printf(F("M AppWren::isWrenResultOK WREN_RESULT_COMPILE_ERROR\n"));
513  sci->printf(F("VM WREN_ERR Compilation Failed\n"));
514  sci->sendLZ4Message();
515  }
516  return false;
517  case WREN_RESULT_RUNTIME_ERROR:
518  if(sci->requestStartLZ4Message()){
519  sci->printf(F("M AppWren::isWrenResultOK WREN_RESULT_RUNTIME_ERROR\n"));
520  sci->printf(F("VM WREN_WRN [Implementation Warning] call forwarding will be disabled until next script load.\n"));
521  sci->sendLZ4Message();
522  }
523  return false;
524  case WREN_RESULT_SUCCESS:
525  return true;
526  default:
527  if(sci->requestStartLZ4Message()){
528  sci->printf(F("M AppWren::isWrenResultOK UNDEFINED RESULT!\n"));
529  sci->sendLZ4Message();
530  }
531  return false; //should never execute
532  }
