92 bool FatPartition::allocContiguous(uint32_t count, uint32_t* firstCluster) {
93  // flag to save place to start next search
94  bool setStart = true;
95  // start of group
96  uint32_t bgnCluster;
97  // end of group
98  uint32_t endCluster;
99  // Start at cluster after last allocated cluster.
100  endCluster = bgnCluster = m_allocSearchStart + 1;
101 
102  // search the FAT for free clusters
103  while (1) {
104  if (endCluster > m_lastCluster) {
105  // Can't find space.
106  DBG_FAIL_MACRO;
107  goto fail;
108  }
109  uint32_t f;
110  int8_t fg = fatGet(endCluster, &f);
111  if (fg < 0) {
112  DBG_FAIL_MACRO;
113  goto fail;
114  }
115  if (f || fg == 0) {
116  // don't update search start if unallocated clusters before endCluster.
117  if (bgnCluster != endCluster) {
118  setStart = false;
119  }
120  // cluster in use try next cluster as bgnCluster
121  bgnCluster = endCluster + 1;
122  } else if ((endCluster - bgnCluster + 1) == count) {
123  // done - found space
124  break;
125  }
126  endCluster++;
127  }
128  // Remember possible next free cluster.
129  if (setStart) {
130  m_allocSearchStart = endCluster;
131  }

134  DBG_FAIL_MACRO;

136  }
137  // link clusters

144  }

147

158  uint32_t sector;
159  uint32_t next;
160  uint8_t* pc;
161 
162  // error if reserved cluster of beyond FAT
163  if (cluster < 2 || cluster > m_lastCluster) {
164  DBG_FAIL_MACRO;
165  goto fail;
166  }
167 
168  if (fatType() == 32) {
169  sector = m_fatStartSector + (cluster >> (m_bytesPerSectorShift - 2));
170  pc = fatCachePrepare(sector, FsCache::CACHE_FOR_READ);
171  if (!pc) {
172  DBG_FAIL_MACRO;
173  goto fail;
174  }
175  uint16_t offset = (cluster << 2) & m_sectorMask;
176  next = getLe32(pc + offset);
177  } else if (fatType() == 16) {
178  cluster &= 0XFFFF;
179  sector = m_fatStartSector + (cluster >> (m_bytesPerSectorShift - 1) );
180  pc = fatCachePrepare(sector, FsCache::CACHE_FOR_READ);
181  if (!pc) {
182  DBG_FAIL_MACRO;
183  goto fail;
184  }
185  uint16_t offset = (cluster << 1) & m_sectorMask;
186  next = getLe16(pc + offset);
187  } else if (FAT12_SUPPORT && fatType() == 12) {
188  uint16_t index = cluster;
189  index += index >> 1;
190  sector = m_fatStartSector + (index >> m_bytesPerSectorShift);
191  pc = fatCachePrepare(sector, FsCache::CACHE_FOR_READ);
192  if (!pc) {

194  goto fail;
195  }
196  index &= m_sectorMask;
197  uint16_t tmp = pc[index];

201  if (!pc) {

203  goto fail;
204  }
205  index = 0;
206  }
207  tmp |= pc[index] << 8;

209  } else {
