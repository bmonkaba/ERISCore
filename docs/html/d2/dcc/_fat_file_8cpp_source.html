771  // get next cluster from FAT
772  fg = m_vol->fatGet(m_curCluster, &m_curCluster);
773  if (fg < 0) {
774  DBG_FAIL_MACRO;
775  goto fail;
776  }
777  if (fg == 0) {
778  if (isDir()) {
779  break;
780  }
781  DBG_FAIL_MACRO;
782  goto fail;
783  }
784  }
785  }
786  sector = m_vol->clusterStartSector(m_curCluster) + sectorOfCluster;
787  }
788  if (offset != 0 || toRead < m_vol->bytesPerSector()
789  || sector == m_vol->cacheSectorNumber()) {
790  // amount to be read from current sector
791  n = m_vol->bytesPerSector() - offset;
792  if (n > toRead) {
793  n = toRead;
794  }
795  // read sector to cache and copy data to caller
796  pc = m_vol->dataCachePrepare(sector, FsCache::CACHE_FOR_READ);
797  if (!pc) {
798  DBG_FAIL_MACRO;
799  goto fail;
800  }
801  uint8_t* src = pc + offset;
802  memcpy(dst, src, n);
803 #if USE_MULTI_SECTOR_IO
804  } else if (toRead >= 2*m_vol->bytesPerSector()) {
805  uint32_t ns = toRead >> m_vol->bytesPerSectorShift();
806  if (!isRootFixed()) {
807  uint32_t mb = m_vol->sectorsPerCluster() - sectorOfCluster;
808  if (mb < ns) {
809  ns = mb;
810  }
811  }
812  n = ns << m_vol->bytesPerSectorShift();
813  if (!m_vol->cacheSafeRead(sector, dst, ns)) {
814  DBG_FAIL_MACRO;
815  goto fail;
816  }
817 #endif // USE_MULTI_SECTOR_IO
818  } else {
819  // read single sector
820  n = m_vol->bytesPerSector();
821  if (!m_vol->cacheSafeRead(sector, dst)) {
822  DBG_FAIL_MACRO;
823  goto fail;
824  }
825  }
826  dst += n;
827  m_curPosition += n;
828  toRead -= n;
829  }
830  return nbyte - toRead;
831 
832  fail:
833  m_error |= READ_ERROR;
834  return -1;
835 }

1131 //------------------------------------------------------------------------------
1132 bool FatFile::seekSet(uint32_t pos) {
1133  uint32_t nCur;
1134  uint32_t nNew;
1135  uint32_t tmp = m_curCluster;
1136  // error if file not open
1137  if (!isOpen()) {
1138  DBG_FAIL_MACRO;
1139  goto fail;
1140  }
1141  // Optimize O_APPEND writes.
1142  if (pos == m_curPosition) {
1143  return true;
1144  }
1145  if (pos == 0) {
1146  // set position to start of file
1147  m_curCluster = 0;
1148  goto done;
1149  }
1150  if (isFile()) {
1151  if (pos > m_fileSize) {
1152  DBG_FAIL_MACRO;
1153  goto fail;
1154  }
1155  } else if (isRootFixed()) {
1156  if (pos <= FS_DIR_SIZE*m_vol->rootDirEntryCount()) {
1157  goto done;
1158  }
1159  DBG_FAIL_MACRO;
1160  goto fail;
1161  }
1162  // calculate cluster index for new position
1163  nNew = (pos - 1) >> (m_vol->bytesPerClusterShift());
1164 #if USE_FAT_FILE_FLAG_CONTIGUOUS
1165  if (isContiguous()) {
1166  m_curCluster = m_firstCluster + nNew;
1167  goto done;
1168  }
1169 #endif // USE_FAT_FILE_FLAG_CONTIGUOUS
1170  // calculate cluster index for current position
1171  nCur = (m_curPosition - 1) >> (m_vol->bytesPerClusterShift());
1172 
1173  if (nNew < nCur || m_curPosition == 0) {
1174  // must follow chain from first cluster
1175  m_curCluster = isRoot32() ? m_vol->rootDirStart() : m_firstCluster;
1176  } else {
1177  // advance from curPosition
1178  nNew -= nCur;
1179  }
1180  while (nNew--) {
1181  if (m_vol->fatGet(m_curCluster, &m_curCluster) <= 0) {
1182  DBG_FAIL_MACRO;
1183  goto fail;
1184  }
1185  }
1186 
1187  done:
1188  m_curPosition = pos;
1189  m_flags &= ~FILE_FLAG_PREALLOCATE;
1190  return true;
1191 
1192  fail:
1193  m_curCluster = tmp;
1194  return false;
1195 }

1331 //------------------------------------------------------------------------------
1332 size_t FatFile::write(const void* buf, size_t nbyte) {
1333  // convert void* to uint8_t* - must be before goto statements
1334  const uint8_t* src = reinterpret_cast<const uint8_t*>(buf);
1335  uint8_t* pc;
1336  uint8_t cacheOption;
1337  // number of bytes left to write - must be before goto statements
1338  size_t nToWrite = nbyte;
1339  size_t n;
1340  // error if not a normal file or is read-only
1341  if (!isWritable()) {
1342  DBG_FAIL_MACRO;
1343  goto fail;
1344  }
1345  // seek to end of file if append flag
1346  if ((m_flags & FILE_FLAG_APPEND)) {
1347  if (!seekSet(m_fileSize)) {
1348  DBG_FAIL_MACRO;
1349  goto fail;
1350  }
1351  }
1352  // Don't exceed max fileSize.
1353  if (nbyte > (0XFFFFFFFF - m_curPosition)) {
1354  DBG_FAIL_MACRO;
1355  goto fail;
1356  }
1357  while (nToWrite) {
1358  uint8_t sectorOfCluster = m_vol->sectorOfCluster(m_curPosition);
1359  uint16_t sectorOffset = m_curPosition & m_vol->sectorMask();
1360  if (sectorOfCluster == 0 && sectorOffset == 0) {
1361  // start of new cluster
1362  if (m_curCluster != 0) {
1363 #if USE_FAT_FILE_FLAG_CONTIGUOUS
1364  int8_t fg;
1365  if (isContiguous() && m_fileSize > m_curPosition) {
1366  m_curCluster++;
1367  fg = 1;
1368  } else {
1369  fg = m_vol->fatGet(m_curCluster, &m_curCluster);
1370  if (fg < 0) {
1371  DBG_FAIL_MACRO;
1372  goto fail;
1373  }
1374  }
1375 #else // USE_FAT_FILE_FLAG_CONTIGUOUS
1376  int8_t fg = m_vol->fatGet(m_curCluster, &m_curCluster);
1377  if (fg < 0) {
1378  DBG_FAIL_MACRO;
1379  goto fail;
1380  }
1381 #endif // USE_FAT_FILE_FLAG_CONTIGUOUS
1382  if (fg == 0) {
1383  // add cluster if at end of chain
1384  if (!addCluster()) {
1385  DBG_FAIL_MACRO;
1386  goto fail;
1387  }
1388  }
1389  } else {
1390  if (m_firstCluster == 0) {
1391  // allocate first cluster of file
1392  if (!addCluster()) {
1393  DBG_FAIL_MACRO;
1394  goto fail;
1395  }
1396  m_firstCluster = m_curCluster;
1397  } else {
1398  m_curCluster = m_firstCluster;
1399  }
1400  }
1401  }
1402  // sector for data write
1403  uint32_t sector = m_vol->clusterStartSector(m_curCluster)
1404  + sectorOfCluster;
1405 
1406  if (sectorOffset != 0 || nToWrite < m_vol->bytesPerSector()) {
1407  // partial sector - must use cache
1408  // max space in sector
1409  n = m_vol->bytesPerSector() - sectorOffset;
1410  // lesser of space and amount to write
1411  if (n > nToWrite) {
1412  n = nToWrite;
1413  }
1414 
1415  if (sectorOffset == 0 &&
1416  (m_curPosition >= m_fileSize || m_flags & FILE_FLAG_PREALLOCATE)) {
1417  // start of new sector don't need to read into cache
1418  cacheOption = FsCache::CACHE_RESERVE_FOR_WRITE;
1419  } else {
1420  // rewrite part of sector
1421  cacheOption = FsCache::CACHE_FOR_WRITE;
1422  }
1423  pc = m_vol->dataCachePrepare(sector, cacheOption);
1424  if (!pc) {
1425  DBG_FAIL_MACRO;
1426  goto fail;
1427  }
1428  uint8_t* dst = pc + sectorOffset;
1429  memcpy(dst, src, n);
1430  if (m_vol->bytesPerSector() == (n + sectorOffset)) {
1431  // Force write if sector is full - improves large writes.
1432  if (!m_vol->cacheSyncData()) {
1433  DBG_FAIL_MACRO;
1434  goto fail;
1435  }
1436  }
1437 #if USE_MULTI_SECTOR_IO
1438  } else if (nToWrite >= 2*m_vol->bytesPerSector()) {
1439  // use multiple sector write command
1440  uint32_t maxSectors = m_vol->sectorsPerCluster() - sectorOfCluster;
1441  uint32_t nSector = nToWrite >> m_vol->bytesPerSectorShift();
1442  if (nSector > maxSectors) {
1443  nSector = maxSectors;
1444  }
1445  n = nSector << m_vol->bytesPerSectorShift();
1446  if (!m_vol->cacheSafeWrite(sector, src, nSector)) {
1447  DBG_FAIL_MACRO;
1448  goto fail;
1449  }
1450 #endif // USE_MULTI_SECTOR_IO
1451  } else {
1452  // use single sector write command
1453  n = m_vol->bytesPerSector();
1454  if (!m_vol->cacheSafeWrite(sector, src)) {
1455  DBG_FAIL_MACRO;
1456  goto fail;
1457  }
1458  }
1459  m_curPosition += n;
1460  src += n;
1461  nToWrite -= n;
1462  }
1463  if (m_curPosition > m_fileSize) {
1464  // update fileSize and insure sync will update dir entry
1465  m_fileSize = m_curPosition;
