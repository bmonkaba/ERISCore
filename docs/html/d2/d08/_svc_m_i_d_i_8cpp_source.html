83 void handleClockCallback(){
84  unsigned long t = micros();
85  last_midi_clock_rxtime = t;
86  if(midi_clock_counter%2) usbMIDI.sendRealTime(midi::Clock);
87  midi_clock_counter += 1;
88  if(midi_clock_counter % 3 == 0) thirtysecond_note_count++;
89  if(midi_clock_counter % 6 == 0) sixteenth_note_count++;
90  if(midi_clock_counter % 12 == 0) {eighth_note_count++; last_eighth_note_time = t; eighth_note_tick ^= true;}
91  if(midi_clock_counter % 24 == 0){
92  quarter_note_tick ^= true;
93  midi_clock_counter = 0;
94  tempo = round(60.0 * (1.0 / ((float)(t - last_quarter_note_time) / 1000000.0)));
95  if (tempo > 240.0) tempo = 240.0;
96  AppManager* am = AppManager::getInstance();
97  am->data->update("TEMPO",tempo);
98  last_quarter_note_time = t;
99  quarter_note_count += 1;
100  }
101 }
