132 bool SharedSpiCard::begin(SdSpiConfig spiConfig) {
133  Timeout timeout;
134  m_spiActive = false;
135  m_errorCode = SD_CARD_ERROR_NONE;
136  m_type = 0;
137  m_csPin = spiConfig.csPin;
138 #if SPI_DRIVER_SELECT >= 2
139  m_spiDriverPtr = spiConfig.spiPort;
140  if (!m_spiDriverPtr) {
141  error(SD_CARD_ERROR_INVALID_CARD_CONFIG);
142  goto fail;
143  }
144 #endif // SPI_DRIVER_SELECT
145  sdCsInit(m_csPin);
146  spiUnselect();
147  spiSetSckSpeed(1000UL*SD_MAX_INIT_RATE_KHZ);
148  spiBegin(spiConfig);
149  uint32_t arg;
150  m_state = IDLE_STATE;
151  spiStart();
152 
153  // must supply min of 74 clock cycles with CS high.
154  spiUnselect();
155  for (uint8_t i = 0; i < 10; i++) {
156  spiSend(0XFF);
157  }
158  spiSelect();
159  // command to go idle in SPI mode
160  for (uint8_t i = 1;; i++) {
161  if (cardCommand(CMD0, 0) == R1_IDLE_STATE) {
162  break;
163  }
164  if (i == SD_CMD0_RETRY) {
165  error(SD_CARD_ERROR_CMD0);
166  goto fail;
167  }
168  }
169 #if USE_SD_CRC
170  if (cardCommand(CMD59, 1) != R1_IDLE_STATE) {
171  error(SD_CARD_ERROR_CMD59);
172  goto fail;
173  }
174 #endif // USE_SD_CRC
175 
176  // check SD version
177  if (!(cardCommand(CMD8, 0x1AA) & R1_ILLEGAL_COMMAND)) {
178  type(SD_CARD_TYPE_SD2);
179  for (uint8_t i = 0; i < 4; i++) {
180  m_status = spiReceive();
181  }
182  if (m_status != 0XAA) {
183  error(SD_CARD_ERROR_CMD8);
184  goto fail;
185  }
186  } else {
187  type(SD_CARD_TYPE_SD1);
188  }

191  timeout.set(SD_INIT_TIMEOUT);
192  while (cardAcmd(ACMD41, arg) != R1_READY_STATE) {
193  // check for timeout

198  }

224 uint8_t SharedSpiCard::cardCommand(uint8_t cmd, uint32_t arg) {
225  if (!syncDevice()) {
226  return 0XFF;
227  }
228  // select card
229  if (!m_spiActive) {
230  spiStart();
231  }
232  if (cmd != CMD12) {
233  if (!waitReady(SD_CMD_TIMEOUT) && cmd != CMD0) {
234  return 0XFF;
235  }
236  }
237 
238 #if USE_SD_CRC
239  // form message
240  uint8_t buf[6];
241  buf[0] = (uint8_t)0x40U | cmd;
242  buf[1] = (uint8_t)(arg >> 24U);
243  buf[2] = (uint8_t)(arg >> 16U);
244  buf[3] = (uint8_t)(arg >> 8U);
245  buf[4] = (uint8_t)arg;
246 
247  // add CRC
248  buf[5] = CRC7(buf, 5);
249 
250  // send message
251  spiSend(buf, 6);
252 #else // USE_SD_CRC
253  // send command
254  spiSend(cmd | 0x40);
255 
256  // send argument
257  uint8_t* pa = reinterpret_cast<uint8_t*>(&arg);
258  for (int8_t i = 3; i >= 0; i--) {
259  spiSend(pa[i]);
260  }
261 
262  // send CRC - correct for CMD0 with arg zero or CMD8 with arg 0X1AA
263  spiSend(cmd == CMD0 ? 0X95 : 0X87);
264 #endif // USE_SD_CRC
265 
266  // discard first fill byte to avoid MISO pull-up problem.
267  spiReceive();
268 
269  // there are 1-8 fill bytes before response. fill bytes should be 0XFF.
270  uint16_t n = 0;
271  do {
272  m_status = spiReceive();
273  } while (m_status & 0X80 && ++n < 10);
274  return m_status;
275 }

277 bool SharedSpiCard::erase(uint32_t firstSector, uint32_t lastSector) {
278  csd_t csd;
279  if (!readCSD(&csd)) {
280  goto fail;
281  }
282  // check for single sector erase
283  if (!csd.v1.erase_blk_en) {
284  // erase size mask
285  uint8_t m = (csd.v1.sector_size_high << 1) | csd.v1.sector_size_low;
286  if ((firstSector & m) != 0 || ((lastSector + 1) & m) != 0) {
287  // error card can't erase specified area
288  error(SD_CARD_ERROR_ERASE_SINGLE_SECTOR);
289  goto fail;
290  }
291  }
292  if (m_type != SD_CARD_TYPE_SDHC) {
293  firstSector <<= 9;
294  lastSector <<= 9;
295  }
296  if (cardCommand(CMD32, firstSector)
297  || cardCommand(CMD33, lastSector)
298  || cardCommand(CMD38, 0)) {
299  error(SD_CARD_ERROR_ERASE);
300  goto fail;
301  }
302  if (!waitReady(SD_ERASE_TIMEOUT)) {
303  error(SD_CARD_ERROR_ERASE_TIMEOUT);
304  goto fail;
305  }
306  spiStop();
307  return true;
308 
309  fail:

338 bool SharedSpiCard::readData(uint8_t* dst, size_t count) {
339 #if USE_SD_CRC
340  uint16_t crc;
341 #endif // USE_SD_CRC
342 
343  // wait for start sector token
344  Timeout timeout(SD_READ_TIMEOUT);
345  while ((m_status = spiReceive()) == 0XFF) {
346  if (timeout.timedOut()) {
347  error(SD_CARD_ERROR_READ_TIMEOUT);
348  goto fail;
349  }
350  }
351  if (m_status != DATA_START_SECTOR) {
352  error(SD_CARD_ERROR_READ_TOKEN);
353  goto fail;
354  }
355  // transfer data
356  if ((m_status = spiReceive(dst, count))) {
357  error(SD_CARD_ERROR_DMA);
358  goto fail;
359  }
360 
361 #if USE_SD_CRC
362  // get crc
363  crc = (spiReceive() << 8) | spiReceive();
364  if (crc != CRC_CCITT(dst, count)) {
365  error(SD_CARD_ERROR_READ_CRC);
366  goto fail;
367  }
368 #else // USE_SD_CRC
369  // discard crc
370  spiReceive();
371  spiReceive();
372 #endif // USE_SD_CRC
373  return true;
374 
375  fail:
376  spiStop();
377  return false;
378 }
