437

440 
441  while (SDHC_SYSCTL & SDHC_SYSCTL_RSTA) {
442  }
443 
444  // Set initial SCK rate.

446 
447  enableGPIO(true);

454  NVIC_ENABLE_IRQ(IRQ_SDHC);
455 
456  // Send 80 clocks to card.
457  SDHC_SYSCTL |= SDHC_SYSCTL_INITA;
458  while (SDHC_SYSCTL & SDHC_SYSCTL_INITA) {
459  }

508  SDHC_BLKATTR = SDHC_BLKATTR_BLKCNT(n) | SDHC_BLKATTR_BLKSIZE(512);
509  SDHC_IRQSIGEN = SDHC_IRQSIGEN_MASK;

511  return false;
512  }

515 //------------------------------------------------------------------------------
516 // Read 16 byte CID or CSD register.
517 static bool readReg16(uint32_t xfertyp, void* data) {
518  uint8_t* d = reinterpret_cast<uint8_t*>(data);
519  if (!cardCommand(xfertyp, m_rca)) {
520  return false; // Caller will set errorCode.
521  }
522  uint32_t sr[] = {SDHC_CMDRSP0, SDHC_CMDRSP1, SDHC_CMDRSP2, SDHC_CMDRSP3};
523  for (int i = 0; i < 15; i++) {
524  d[14 - i] = sr[i/4] >> 8*(i%4);
525  }
526  d[15] = 0;
527  return true;
528 }
529 //------------------------------------------------------------------------------
530 static void setSdclk(uint32_t kHzMax) {
531  const uint32_t DVS_LIMIT = 0X10;
532  const uint32_t SDCLKFS_LIMIT = 0X100;
533  uint32_t dvs = 1;
534  uint32_t sdclkfs = 1;
535  uint32_t maxSdclk = 1000*kHzMax;
536  uint32_t base = baseClock();
537 
538  while ((base/(sdclkfs*DVS_LIMIT) > maxSdclk) && (sdclkfs < SDCLKFS_LIMIT)) {
539  sdclkfs <<= 1;
540  }
541  while ((base/(sdclkfs*dvs) > maxSdclk) && (dvs < DVS_LIMIT)) {
542  dvs++;
543  }
544  m_sdClkKhz = base/(1000*sdclkfs*dvs);
545  sdclkfs >>= 1;
546  dvs--;
547 #if defined(__MK64FX512__) || defined(__MK66FX1M0__)
548  // Disable SDHC clock.
549  SDHC_SYSCTL &= ~SDHC_SYSCTL_SDCLKEN;
550 #endif // defined(__MK64FX512__) || defined(__MK66FX1M0__)
551 
552  // Change dividers.
553  uint32_t sysctl = SDHC_SYSCTL & ~(SDHC_SYSCTL_DTOCV_MASK
554  | SDHC_SYSCTL_DVS_MASK | SDHC_SYSCTL_SDCLKFS_MASK);
555 
556  SDHC_SYSCTL = sysctl | SDHC_SYSCTL_DTOCV(0x0E) | SDHC_SYSCTL_DVS(dvs)
557  | SDHC_SYSCTL_SDCLKFS(sdclkfs);
558 
559  // Wait until the SDHC clock is stable.
560  while (!(SDHC_PRSSTAT & SDHC_PRSSTAT_SDSTB)) {
561  }
562 
563 #if defined(__MK64FX512__) || defined(__MK66FX1M0__)
564  // Enable the SDHC clock.
565  SDHC_SYSCTL |= SDHC_SYSCTL_SDCLKEN;
566 #endif // defined(__MK64FX512__) || defined(__MK66FX1M0__)
567 }

569 static bool transferStop() {
570  // This fix allows CDIHB to be cleared in Tennsy 3.x without a reset.
571  SDHC_PROCTL &= ~SDHC_PROCTL_SABGREQ;
572  if (!cardCommand(CMD12_XFERTYP, 0)) {
573  return sdError(SD_CARD_ERROR_CMD12);
574  }
575 // if (yieldTimeout(isBusyCMD13)) {
576  if (yieldTimeout(isBusyDat)) {
577  return sdError(SD_CARD_ERROR_CMD13);
578  }
579  if (SDHC_PRSSTAT & SDHC_PRSSTAT_CDIHB) {
580  // This should not happen after above fix.
581  // Save registers before reset DAT lines.
582  uint32_t irqsststen = SDHC_IRQSTATEN;
583  uint32_t proctl = SDHC_PROCTL & ~SDHC_PROCTL_SABGREQ;
584  // Do reset to clear CDIHB. Should be a better way!
585  SDHC_SYSCTL |= SDHC_SYSCTL_RSTD;
586  // Restore registers.

592 //------------------------------------------------------------------------------
593 // Return true if timeout occurs.
594 static bool yieldTimeout(bool (*fcn)()) {
595  m_busyFcn = fcn;
596  uint32_t m = micros();

599  m_busyFcn = 0;

642 bool SdioCard::begin(SdioConfig sdioConfig) {
643  uint32_t kHzSdClk;
644  uint32_t arg;
645  m_sdioConfig = sdioConfig;
646  m_curState = IDLE_STATE;
647  m_initDone = false;
648  m_errorCode = SD_CARD_ERROR_NONE;
649  m_highCapacity = false;
650  m_version2 = false;
651 
652  // initialize controller.
653  initSDHC();
654  if (!cardCommand(CMD0_XFERTYP, 0)) {
655  return sdError(SD_CARD_ERROR_CMD0);
656  }
657  // Try several times for case of reset delay.
658  for (uint32_t i = 0; i < CMD8_RETRIES; i++) {
659  if (cardCommand(CMD8_XFERTYP, 0X1AA)) {
660  if (SDHC_CMDRSP0 != 0X1AA) {
661  return sdError(SD_CARD_ERROR_CMD8);
662  }
663  m_version2 = true;
664  break;
665  }
666  }
667  arg = m_version2 ? 0X40300000 : 0x00300000;
668  int m = micros();
669  do {
670  if (!cardAcmd(0, ACMD41_XFERTYP, arg) ||
671  ((micros() - m) > BUSY_TIMEOUT_MICROS)) {
672  return sdError(SD_CARD_ERROR_ACMD41);
673  }
674  } while ((SDHC_CMDRSP0 & 0x80000000) == 0);
675  m_ocr = SDHC_CMDRSP0;
676  if (SDHC_CMDRSP0 & 0x40000000) {
677  // Is high capacity.
678  m_highCapacity = true;
679  }
680  if (!cardCommand(CMD2_XFERTYP, 0)) {
681  return sdError(SD_CARD_ERROR_CMD2);
682  }
683  if (!cardCommand(CMD3_XFERTYP, 0)) {
684  return sdError(SD_CARD_ERROR_CMD3);
685  }
686  m_rca = SDHC_CMDRSP0 & 0xFFFF0000;
687 
688  if (!readReg16(CMD9_XFERTYP, &m_csd)) {
689  return sdError(SD_CARD_ERROR_CMD9);
690  }
691  if (!readReg16(CMD10_XFERTYP, &m_cid)) {
692  return sdError(SD_CARD_ERROR_CMD10);
693  }
694  if (!cardCommand(CMD7_XFERTYP, m_rca)) {
695  return sdError(SD_CARD_ERROR_CMD7);
696  }
697  // Set card to bus width four.
698  if (!cardAcmd(m_rca, ACMD6_XFERTYP, 2)) {
699  return sdError(SD_CARD_ERROR_ACMD6);
700  }
701  // Set SDHC to bus width four.
702  SDHC_PROCTL &= ~SDHC_PROCTL_DTW_MASK;
703  SDHC_PROCTL |= SDHC_PROCTL_DTW(SDHC_PROCTL_DTW_4BIT);
704 
705  SDHC_WML = SDHC_WML_RDWML(FIFO_WML) | SDHC_WML_WRWML(FIFO_WML);
706 
707  // Determine if High Speed mode is supported and set frequency.
708  // Check status[16] for error 0XF or status[16] for new mode 0X1.
709  uint8_t status[64];
710  if (cardCMD6(0X00FFFFFF, status) && (2 & status[13]) &&
711  cardCMD6(0X80FFFFF1, status) && (status[16] & 0XF) == 1) {
712  kHzSdClk = 50000;
713  } else {
714  kHzSdClk = 25000;
715  }
716  // Disable GPIO.
717  enableGPIO(false);
718 
719  // Set the SDHC SCK frequency.
720  setSdclk(kHzSdClk);
721 
722  // Enable GPIO.
723  enableGPIO(true);
724  m_initDone = true;
725  return true;
726 }
