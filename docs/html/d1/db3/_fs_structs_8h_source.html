102 }
103 //------------------------------------------------------------------------------
104 // Reserved characters for FAT short 8.3 names.
105 inline bool sfnReservedChar(uint8_t c) {
106  if (c == '"' || c == '|' || c == '[' || c == '\\' || c == ']') {
107  return true;
108  }
109  // *+,./ or :;<=>?
110  if ((0X2A <= c && c <= 0X2F && c != 0X2D) || (0X3A <= c && c <= 0X3F)) {
111  return true;
112  }
113  // Reserved if not in range (0X20, 0X7F).
114  return !(0X20 < c && c < 0X7F);
115 }
