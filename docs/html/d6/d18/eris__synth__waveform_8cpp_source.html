47 void AudioSynthWaveform2k::update(void)
48 {
49  audio_block_t *block;
50  int16_t *bp, *end;
51  int32_t val1, val2;
52  int16_t magnitude15;
53  uint32_t i, ph, index, index2, scale;
54  const uint32_t inc = phase_increment;
55 
56  ph = phase_accumulator + phase_offset;
57  if (magnitude == 0) {
58  phase_accumulator += inc * AUDIO_BLOCK_SAMPLES;
59  return;
60  }
61  block = allocate();
62  if (!block) {
63  phase_accumulator += inc * AUDIO_BLOCK_SAMPLES;
64  return;
65  }
66  bp = block->data;
67 
68  switch(tone_type) {
69  case WAVEFORM_SINE:
70  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
71  index = ph >> 21;
72  val1 = AudioWaveformSine[index];
73  val2 = AudioWaveformSine[index+1];
74  scale = (ph >> 8) & 0xFFFF;
75  val2 *= scale;
76  val1 *= 0x10000 - scale;
77  *bp++ = multiply_32x32_rshift32(val1 + val2, magnitude);
78  ph += inc;
79  }
80  break;
81 
82  case WAVEFORM_ARBITRARY:
83  if (!arbdata) {
84  release(block);
85  phase_accumulator += inc * AUDIO_BLOCK_SAMPLES;
86  return;
87  }
88  // len = 256
89  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
90  index = ph >> 21;
91  index2 = index + 1;
92  if (index2 >= 256) index2 = 0;
93  val1 = *(arbdata + index);
94  val2 = *(arbdata + index2);
95  scale = (ph >> 8) & 0xFFFF;
96  val2 *= scale;
97  val1 *= 0x10000 - scale;
98  *bp++ = multiply_32x32_rshift32(val1 + val2, magnitude);
99  ph += inc;
100  }
101  break;
102 
103  case WAVEFORM_SQUARE:
104  magnitude15 = signed_saturate_rshift(magnitude, 16, 1);
105  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
106  if (ph & 0x80000000) {
107  *bp++ = -magnitude15;
108  } else {
109  *bp++ = magnitude15;
110  }
111  ph += inc;
112  }
113  break;
114 
115  case WAVEFORM_BANDLIMIT_SQUARE:
116  for (int i = 0 ; i < AUDIO_BLOCK_SAMPLES ; i++)
117  {
118  uint32_t new_ph = ph + inc ;
119  int16_t val = band_limit_waveform.generate_square (new_ph, i) ;
120  *bp++ = (val * magnitude) >> 16 ;
121  ph = new_ph ;
122  }
123  break;
124 
125  case WAVEFORM_SAWTOOTH:
126  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
127  *bp++ = signed_multiply_32x16t(magnitude, ph);
128  ph += inc;
129  }
130  break;
131 
132  case WAVEFORM_SAWTOOTH_REVERSE:
133  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
134  *bp++ = signed_multiply_32x16t(0xFFFFFFFFu - magnitude, ph);
135  ph += inc;
136  }
137  break;
138 
139  case WAVEFORM_BANDLIMIT_SAWTOOTH:
140  case WAVEFORM_BANDLIMIT_SAWTOOTH_REVERSE:
141  for (i = 0 ; i < AUDIO_BLOCK_SAMPLES; i++)
142  {
143  uint32_t new_ph = ph + inc ;
144  int16_t val = band_limit_waveform.generate_sawtooth (new_ph, i) ;
145  if (tone_type == WAVEFORM_BANDLIMIT_SAWTOOTH_REVERSE)
146  *bp++ = (val * -magnitude) >> 16 ;
147  else
148  *bp++ = (val * magnitude) >> 16 ;
149  ph = new_ph ;
150  }
151  break;
152 
153  case WAVEFORM_TRIANGLE:
154  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
155  uint32_t phtop = ph >> 30;
156  if (phtop == 1 || phtop == 2) {
157  *bp++ = ((0xFFFF - (ph >> 15)) * magnitude) >> 16;
158  } else {
159  *bp++ = (((int32_t)ph >> 15) * magnitude) >> 16;
160  }
161  ph += inc;
162  }
163  break;
164 
165  case WAVEFORM_TRIANGLE_VARIABLE:
166  do {
167  uint32_t rise = 0xFFFFFFFF / (pulse_width >> 16);
168  uint32_t fall = 0xFFFFFFFF / (0xFFFF - (pulse_width >> 16));
169  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
170  if (ph < pulse_width/2) {
171  uint32_t n = (ph >> 16) * rise;
172  *bp++ = ((n >> 16) * magnitude) >> 16;
173  } else if (ph < 0xFFFFFFFF - pulse_width/2) {
174  uint32_t n = 0x7FFFFFFF - (((ph - pulse_width/2) >> 16) * fall);
175  *bp++ = (((int32_t)n >> 16) * magnitude) >> 16;
176  } else {
177  uint32_t n = ((ph + pulse_width/2) >> 16) * rise + 0x80000000;
178  *bp++ = (((int32_t)n >> 16) * magnitude) >> 16;
179  }
180  ph += inc;
181  }
182  } while (0);
183  break;
184 
185  case WAVEFORM_PULSE:
186  magnitude15 = signed_saturate_rshift(magnitude, 16, 1);
187  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
188  if (ph < pulse_width) {
189  *bp++ = magnitude15;
190  } else {
191  *bp++ = -magnitude15;
192  }
193  ph += inc;
194  }
195  break;
196 
197  case WAVEFORM_BANDLIMIT_PULSE:
198  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++)
199  {
200  int32_t new_ph = ph + inc ;
201  int32_t val = band_limit_waveform.generate_pulse (new_ph, pulse_width, i) ;
202  *bp++ = (int16_t) ((val * magnitude) >> 16) ;
203  ph = new_ph ;
204  }
205  break;
206 
207  case WAVEFORM_SAMPLE_HOLD:
208  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
209  *bp++ = sample;
210  uint32_t newph = ph + inc;
211  if (newph < ph) {
212  sample = random(magnitude) - (magnitude >> 1);
213  }
214  ph = newph;
215  }
216  break;
217  }
218  phase_accumulator = ph - phase_offset;
219 
220  if (tone_offset) {
221  bp = block->data;
222  end = bp + AUDIO_BLOCK_SAMPLES;
223  do {
224  val1 = *bp;
225  *bp++ = signed_saturate_rshift(val1 + tone_offset, 16, 0);
226  } while (bp < end);
227  }
228  transmit(block, 0);
229  release(block);
230 }

241 void AudioSynthWaveformModulated2k::update(void)
242 {
243  audio_block_t *block, *moddata, *shapedata;
244  int16_t *bp, *end;
245  int32_t val1, val2;
246  int16_t magnitude15;
247  uint32_t i, ph, index, index2, scale, priorphase;
248  const uint32_t inc = phase_increment;
249 
250  moddata = receiveReadOnly(0);
251  shapedata = receiveReadOnly(1);
252 
253  // Pre-compute the phase angle for every output sample of this update
254  ph = phase_accumulator;
255  priorphase = phasedata[AUDIO_BLOCK_SAMPLES-1];
256  if (moddata && modulation_type == 0) {
257  // Frequency Modulation
258  bp = moddata->data;
259  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
260  int32_t n = (*bp++) * modulation_factor; // n is # of octaves to mod
261  int32_t ipart = n >> 27; // 4 integer bits
262  n &= 0x7FFFFFF; // 27 fractional bits
263  #ifdef IMPROVE_EXPONENTIAL_ACCURACY
264  // exp2 polynomial suggested by Stefan Stenzel on "music-dsp"
265  // mail list, Wed, 3 Sep 2014 10:08:55 +0200
266  int32_t x = n << 3;
267  n = multiply_accumulate_32x32_rshift32_rounded(536870912, x, 1494202713);
268  int32_t sq = multiply_32x32_rshift32_rounded(x, x);
269  n = multiply_accumulate_32x32_rshift32_rounded(n, sq, 1934101615);
270  n = n + (multiply_32x32_rshift32_rounded(sq,
271  multiply_32x32_rshift32_rounded(x, 1358044250)) << 1);
272  n = n << 1;
273  #else
274  // exp2 algorithm by Laurent de Soras
275  // https://www.musicdsp.org/en/latest/Other/106-fast-exp2-approximation.html
276  n = (n + 134217728) << 3;
277 
278  n = multiply_32x32_rshift32_rounded(n, n);
279  n = multiply_32x32_rshift32_rounded(n, 715827883) << 3;
280  n = n + 715827882;
281  #endif
282  uint32_t scale = n >> (14 - ipart);
283  uint64_t phstep = (uint64_t)inc * scale;
284  uint32_t phstep_msw = phstep >> 32;
285  if (phstep_msw < 0x7FFE) {
286  ph += phstep >> 16;
287  } else {
288  ph += 0x7FFE0000;
289  }
290  phasedata[i] = ph;
291  }
292  release(moddata);
293  } else if (moddata) {
294  // Phase Modulation
295  bp = moddata->data;
296  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
297  // more than +/- 180 deg shift by 32 bit overflow of "n"
298  uint32_t n = ((uint32_t)(*bp++)) * modulation_factor;
299  phasedata[i] = ph + n;
300  ph += inc;
301  }
302  release(moddata);
303  } else {
304  // No Modulation Input
305  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
306  phasedata[i] = ph;
307  ph += inc;
308  }
309  }
310  phase_accumulator = ph;
311 
312  // If the amplitude is zero, no output, but phase still increments properly
313  if (magnitude == 0) {
314  if (shapedata) release(shapedata);
315  return;
316  }
317  block = allocate();
318  if (!block) {
319  if (shapedata) release(shapedata);
320  return;
321  }
322  bp = block->data;
323 
324  // Now generate the output samples using the pre-computed phase angles
325  switch(tone_type) {
326  case WAVEFORM_SINE:
327  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
328  ph = phasedata[i];
329  index = ph >> 21;
330  val1 = AudioWaveformSine[index];
331  val2 = AudioWaveformSine[index+1];
332  scale = (ph >> 8) & 0xFFFF;
333  val2 *= scale;
334  val1 *= 0x10000 - scale;
335  *bp++ = multiply_32x32_rshift32(val1 + val2, magnitude);
336  }
337  break;
338 
339  case WAVEFORM_ARBITRARY:
340  if (!arbdata) {
341  release(block);
342  if (shapedata) release(shapedata);
343  return;
344  }
345  // len = 256
346  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
347  ph = phasedata[i];
348  index = ph >> 21;//ERIS CORE -- change from 24 (256 sample) to 21 (2048 samples)
349  index2 = index + 1;
350  if (index2 >= 256) index2 = 0;
351  val1 = *(arbdata + index);
352  val2 = *(arbdata + index2);
353  scale = (ph >> 8) & 0xFFFF;
354  val2 *= scale;
355  val1 *= 0x10000 - scale;
356  *bp++ = multiply_32x32_rshift32(val1 + val2, magnitude);
357  }
358  break;
359 
360  case WAVEFORM_PULSE:
361  if (shapedata) {
362  magnitude15 = signed_saturate_rshift(magnitude, 16, 1);
363  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
364  uint32_t width = ((shapedata->data[i] + 0x8000) & 0xFFFF) << 16;
365  if (phasedata[i] < width) {
366  *bp++ = magnitude15;
367  } else {
368  *bp++ = -magnitude15;
369  }
370  }
371  break;
372  } // else fall through to orginary square without shape modulation
373 
374  case WAVEFORM_SQUARE:
375  magnitude15 = signed_saturate_rshift(magnitude, 16, 1);
376  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
377  if (phasedata[i] & 0x80000000) {
378  *bp++ = -magnitude15;
379  } else {
380  *bp++ = magnitude15;
381  }
382  }
383  break;
384 
385  case WAVEFORM_BANDLIMIT_PULSE:
386  if (shapedata)
387  {
388  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++)
389  {
390  uint32_t width = ((shapedata->data[i] + 0x8000) & 0xFFFF) << 16;
391  int32_t val = band_limit_waveform.generate_pulse (phasedata[i], width, i) ;
392  *bp++ = (int16_t) ((val * magnitude) >> 16) ;
393  }
394  break;
395  } // else fall through to orginary square without shape modulation
396 
397  case WAVEFORM_BANDLIMIT_SQUARE:
398  for (i = 0 ; i < AUDIO_BLOCK_SAMPLES ; i++)
399  {
400  int32_t val = band_limit_waveform.generate_square (phasedata[i], i) ;
401  *bp++ = (int16_t) ((val * magnitude) >> 16) ;
402  }
403  break;
404 
405  case WAVEFORM_SAWTOOTH:
406  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
407  *bp++ = signed_multiply_32x16t(magnitude, phasedata[i]);
408  }
409  break;
410 
411  case WAVEFORM_SAWTOOTH_REVERSE:
412  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
413  *bp++ = signed_multiply_32x16t(0xFFFFFFFFu - magnitude, phasedata[i]);
414  }
415  break;
416 
417  case WAVEFORM_BANDLIMIT_SAWTOOTH:
418  case WAVEFORM_BANDLIMIT_SAWTOOTH_REVERSE:
419  for (i = 0 ; i < AUDIO_BLOCK_SAMPLES ; i++)
420  {
421  int16_t val = band_limit_waveform.generate_sawtooth (phasedata[i], i) ;
422  val = (int16_t) ((val * magnitude) >> 16) ;
423  *bp++ = tone_type == WAVEFORM_BANDLIMIT_SAWTOOTH_REVERSE ? (int16_t) -val : (int16_t) +val ;
424  }
425  break;
426 
427  case WAVEFORM_TRIANGLE_VARIABLE:
428  if (shapedata) {
429  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
430  uint32_t width = (shapedata->data[i] + 0x8000) & 0xFFFF;
431  uint32_t rise = 0xFFFFFFFF / width;
432  uint32_t fall = 0xFFFFFFFF / (0xFFFF - width);
433  uint32_t halfwidth = width << 15;
434  uint32_t n;
435  ph = phasedata[i];
436  if (ph < halfwidth) {
437  n = (ph >> 16) * rise;
438  *bp++ = ((n >> 16) * magnitude) >> 16;
439  } else if (ph < 0xFFFFFFFF - halfwidth) {
440  n = 0x7FFFFFFF - (((ph - halfwidth) >> 16) * fall);
441  *bp++ = (((int32_t)n >> 16) * magnitude) >> 16;
442  } else {
443  n = ((ph + halfwidth) >> 16) * rise + 0x80000000;
444  *bp++ = (((int32_t)n >> 16) * magnitude) >> 16;
445  }
446  ph += inc;
447  }
448  break;
449  } // else fall through to orginary triangle without shape modulation
450 
451  case WAVEFORM_TRIANGLE:
452  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
453  ph = phasedata[i];
454  uint32_t phtop = ph >> 30;
455  if (phtop == 1 || phtop == 2) {
456  *bp++ = ((0xFFFF - (ph >> 15)) * magnitude) >> 16;
457  } else {
458  *bp++ = (((int32_t)ph >> 15) * magnitude) >> 16;
459  }
460  }
461  break;
462  case WAVEFORM_SAMPLE_HOLD:
463  for (i=0; i < AUDIO_BLOCK_SAMPLES; i++) {
464  ph = phasedata[i];
465  if (ph < priorphase) { // does not work for phase modulation
466  sample = random(magnitude) - (magnitude >> 1);
467  }
468  priorphase = ph;
469  *bp++ = sample;
470  }
471  break;
472  }
473 
474  if (tone_offset) {
475  bp = block->data;
476  end = bp + AUDIO_BLOCK_SAMPLES;
477  do {
478  val1 = *bp;
479  *bp++ = signed_saturate_rshift(val1 + tone_offset, 16, 0);
480  } while (bp < end);
481  }
482  if (shapedata) release(shapedata);
483  transmit(block, 0);
484  release(block);
485 }
