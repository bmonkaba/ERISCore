56  if (m_capacityMB <= 6) {

66  m_sectorsPerCluster = 16;

132 bool FatFormatter::makeFat16() {
133  uint32_t nc;
134  uint32_t r;
135  PbsFat_t* pbs = reinterpret_cast<PbsFat_t*>(m_secBuf);
136 
137  for (m_dataStart = 2*BU16; ; m_dataStart += BU16) {
138  nc = (m_sectorCount - m_dataStart)/m_sectorsPerCluster;
139  m_fatSize = (nc + 2 + (BYTES_PER_SECTOR/2) - 1)/(BYTES_PER_SECTOR/2);
140  r = BU16 + 1 + 2*m_fatSize + FAT16_ROOT_SECTOR_COUNT;
141  if (m_dataStart >= r) {
142  m_relativeSectors = m_dataStart - r + BU16;
143  break;
144  }
145  }
146  // check valid cluster count for FAT16 volume
147  if (nc < 4085 || nc >= 65525) {
148  writeMsg("Bad cluster count\r\n");
149  return false;
150  }
151  m_reservedSectorCount = 1;
152  m_fatStart = m_relativeSectors + m_reservedSectorCount;
153  m_totalSectors = nc*m_sectorsPerCluster
154  + 2*m_fatSize + m_reservedSectorCount + 32;
155  if (m_totalSectors < 65536) {
156  m_partType = 0X04;
157  } else {
158  m_partType = 0X06;
159  }
160  // write MBR
161  if (!writeMbr()) {
162  return false;
163  }
164  initPbs();
165  setLe16(pbs->bpb.bpb16.rootDirEntryCount, FAT16_ROOT_ENTRY_COUNT);
166  setLe16(pbs->bpb.bpb16.sectorsPerFat16, m_fatSize);
167  pbs->bpb.bpb16.physicalDriveNumber = 0X80;
168  pbs->bpb.bpb16.extSignature = EXTENDED_BOOT_SIGNATURE;
169  setLe32(pbs->bpb.bpb16.volumeSerialNumber, 1234567);
170  for (size_t i = 0; i < sizeof(pbs->bpb.bpb16.volumeLabel); i++) {
171  pbs->bpb.bpb16.volumeLabel[i] = ' ';
172  }
173  pbs->bpb.bpb16.volumeType[0] = 'F';
174  pbs->bpb.bpb16.volumeType[1] = 'A';
175  pbs->bpb.bpb16.volumeType[2] = 'T';
176  pbs->bpb.bpb16.volumeType[3] = '1';
177  pbs->bpb.bpb16.volumeType[4] = '6';
178  if (!m_dev->writeSector(m_relativeSectors, m_secBuf)) {
179  return false;
180  }
181  return initFatDir(16, m_dataStart - m_fatStart);
