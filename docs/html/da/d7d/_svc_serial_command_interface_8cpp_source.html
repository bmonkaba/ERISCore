169 bool FASTRUN SvcSerialCommandInterface::throttle(){
170  uint16_t avail;
171  avail = Serial.availableForWrite();
172 
173  if(avail < SERIAL_THROTTLE_CHECK_CONNECTION_BUFFER_THRESHOLD){
174  uint16_t delta_avail;
175  uint16_t msec_delay_counter = 0;
176  //serial tx buffer not available
177  do{
178  delay(0.5);
179  msec_delay_counter += 2;
180  delta_avail = Serial.availableForWrite();
181  if (delta_avail > SERIAL_THROTTLE_CHECK_CONNECTION_BUFFER_THRESHOLD) break;
182  } while (msec_delay_counter < SERIAL_THROTTLE_CHECK_CONNECTION_MAX_DELAY);
183 
184  if(avail == delta_avail){
185  return true;
186  } else{
187  //Serial.println(F("M WRN throttling"));//use this for debug only!
188  am->data->increment("SERIAL_THROTTLE_EVENTS");
189  }
190  }
191  if(avail < SERIAL_THROTTLE_BUFFER_THRESHOLD) return true;
192  return false;

646 void FASTRUN SvcSerialCommandInterface::update(){
647  char endMarker = '\n';
648  boolean newRxMsg = false;
649  char bufferChr;
650  am->requestArmSetClock(CPU_BOOST_MAX_FREQ);
651 
652 
653  if (is_streaming_file){ //streaming file handler
654  streamTransmitHandler();
655  }
656 
657  if(is_capturing_bulk_data){ //bulk data state handler
658  streamReceiveHandler();
659  }else { //check for incomming data
660  //This is the incomming message handler - it collects the message one byte at a time until
661  //the end of message marker. If a full message is received the newRxMsg flag is set true.
662  //if a partial message is received the flag remains false.
663  //if the receiving buffer reaches an overflow condition, reset the buffer and clear in remaining serial input
664  if (Serial.available() > 0 && false == newRxMsg ){
665  do{
666  bufferChr = Serial.read();
667  received_chars[index_rx_buffer++] = bufferChr;
668 
669  if(index_rx_buffer >= (SERIAL_RX_BUFFER_SIZE-2)){
670  //input overflow - clear serial input and reset the index
671  index_rx_buffer = 0;
672  Serial.clear();
673  Serial.clearReadError();
674  am->requestArmSetClock(CPU_LOW_POWER_MODE_FREQ);
675  return;
676  }
677 
678  if (bufferChr == endMarker){
679  received_chars[--index_rx_buffer] = '\0'; //remove the end marker and null terminate the string
680  index_rx_buffer = 0; //reset the input write index
681  newRxMsg = true;
682  }
683  } while (Serial.available() > 0 && false == newRxMsg);
684  }
685  if (newRxMsg){
686  //Once a new message is received, first the command is parsed out from the message
687  // this first split into cmd and param is for the purpose of identifying the requested command
688  // each individual message handler is then responsible for parsing out thier own parameters if required.
689  char cmd[128], param[128],param2[128];
