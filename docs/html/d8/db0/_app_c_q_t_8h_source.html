321  void FLASHMEM updateOscillatorBank(bool low_range_switch){
322  bool found;
323  float peak_read=-1000;
324  float peak;
325 
326  //sort the FFTReadRange array by cqt bin number
327  if (low_range_switch) {erisAudioAnalyzeFFT1024::sort_fftrr_by_cqt_bin(fftLowRR,NOTE_ARRAY_LENGTH);}
328  else erisAudioAnalyzeFFT1024::sort_fftrr_by_cqt_bin(fftHighRR,NOTE_ARRAY_LENGTH);
329  //read the fft ranges, which stores the results into the FFTReadRange array
330  for (uint16_t i=0; i < NOTE_ARRAY_LENGTH; i++){
331  if (low_range_switch) peak = fft2->read(&fftLowRR[i]);
332  else peak = fft->read(&fftHighRR[i]);
333  if (peak>peak_read){
334  peak_read = peak;
335  }
336  }
337  //update the oscillator bank with the current values
338  for (uint16_t i=0; i < osc_bank_size; i++){
339  if (oscBank[i].cqtBin >= high_range){
340  if (low_range_switch==false){
341  oscBank[i] = fftHighRR[oscBank[i].cqtBin];
342  }
343  }else if (low_range_switch==true){
344  oscBank[i] = fftLowRR[oscBank[i].cqtBin];
345  }
346  }
347  //sort the updated cqt bins by peakValue
348  if (low_range_switch){erisAudioAnalyzeFFT1024::sort_fftrr_by_value(fftLowRR,NOTE_ARRAY_LENGTH);
349  } else erisAudioAnalyzeFFT1024::sort_fftrr_by_value(fftHighRR,NOTE_ARRAY_LENGTH);
350 
351  //Add the new osc settings
352  //low range
353  if (low_range_switch){
354  for(uint16_t i=0; i < osc_bank_size; i++){
355  found = false;
356  for (uint16_t j=0; j < osc_bank_size; j++){
357  if (oscBank[j].cqtBin == fftLowRR[i].cqtBin){
358  //oscBank[j] = fftLowRR[i];
359  found = true;break;
360  }
361  }
362  if(!found){
363  for(int16_t k= osc_bank_size-1; k >= 0;k--){
364  if ((oscBank[k].avgValueFast) < (fftLowRR[i].avgValueFast)){
365  //fftLowRR[i].phase = 0;
366  pll_p = 0.0;
367  pll_f = 1.0;
368  oscBank[k] = fftLowRR[i];
369  break;
370  }
371  }
372  }
373  }
374  } else for(uint16_t i=0; i < osc_bank_size; i++){ //high range
375  found = false;
376  for (uint16_t j=0; j < osc_bank_size; j++){
377  if (oscBank[j].cqtBin == fftHighRR[i].cqtBin){
378  //oscBank[j] = fftHighRR[i];
379  found = true;break;
380  }
381  }
382  if(!found){
383  for(int16_t k= osc_bank_size-1; k >= 0;k--){
384  if ((oscBank[k].avgValueFast) < (fftHighRR[i].avgValueFast)){
385  //fftHighRR[i].phase = 0;
386  pll_p = 0.0;
387  pll_f = 1.0;
388  oscBank[k] = fftHighRR[i];
389  break;
390  }
391  }
392  }
393  }
394 
395  pll_p = 0.0;
396  pll_f = 1.0;
397 
398  //sort the updated cqt bins by peakValue
399  if (low_range_switch){erisAudioAnalyzeFFT1024::sort_fftrr_by_value(fftLowRR,NOTE_ARRAY_LENGTH);
400  } else erisAudioAnalyzeFFT1024::sort_fftrr_by_value(fftHighRR,NOTE_ARRAY_LENGTH);
401 
402  //take the phase from the dominant frequency component
403  float dominantPhase = 0;
404  if (low_range_switch){
405  dominantPhase = fftLowRR[0].phase;
406  } else dominantPhase = fftHighRR[0].phase;;
407 
408  //actually update the oscilators -- IF the data is not TOO old

412  float a,f;

415  if (oscBank[i].peakFrequency > 30.0){
416  f = oscBank[i].peakFrequency;
417  a = sqrt(oscBank[i].avgValueFast)/5.0;//(log1pf(oscBank[i].avgValueFast)/(log1pf(osc_bank_size)));
418  if(!isnan(a)){

426  osc[i]->frequency(f);
427  osc[i]->amplitude(a);
428  osc[i]->phase(phase_aligner);
429  }; //else osc[i]->amplitude(0.0);

437  //else erisAudioAnalyzeFFT1024::sort_fftrr_by_cqt_bin(fftHighRR,NOTE_ARRAY_LENGTH);
