227 bool FatFile::parsePathName(const char* path, FatSfn_t* fname,
228  const char** ptr) {
229  uint8_t uc = 0;
230  uint8_t lc = 0;
231  uint8_t bit = FNAME_FLAG_LC_BASE;
232  // blank fill name and extension
233  for (uint8_t i = 0; i < 11; i++) {
234  fname->sfn[i] = ' ';
235  }
236  for (uint8_t i = 0, n = 7;; path++) {
237  uint8_t c = *path;
238  if (c == 0 || isDirSeparator(c)) {
239  // Done.
240  break;
241  }
242  if (c == '.' && n == 7) {
243  n = 10; // max index for full 8.3 name
244  i = 8; // place for extension
245 
246  // bit for extension.
247  bit = FNAME_FLAG_LC_EXT;
248  } else {
249  if (sfnReservedChar(c) || i > n) {
250  DBG_FAIL_MACRO;
251  goto fail;
252  }
253  if ('a' <= c && c <= 'z') {
254  c += 'A' - 'a';
255  lc |= bit;
256  } else if ('A' <= c && c <= 'Z') {
257  uc |= bit;
258  }
259  fname->sfn[i++] = c;
260  }
261  }
262  // must have a file name, extension is optional
263  if (fname->sfn[0] == ' ') {
264  DBG_FAIL_MACRO;
265  goto fail;
266  }
267  // Set base-name and extension bits.
268  fname->flags = lc & uc ? 0 : lc;
269  while (isDirSeparator(*path)) {
270  path++;
271  }
272  *ptr = path;
