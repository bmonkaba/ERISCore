145  monitor_update = (monitor_dd_update_timer > APPMANAGER_MONITOR_DD_UPDATE_RATE_MSEC);
146  cycle_time=0;
147  drt = display_refresh_time;
148  touch_updated = false;
149  node = root;
150 
151  requestArmSetClock(CPU_BASE_FREQ);
152 
153  if (node == 0){
154  Serial.println(F("AppManager::update called without an application initalized"));
155  return;
156  }
157 
158  //update analog inputs
159  update_analog = analog.update();
160 
161  //update loop is directed by the state variable
162  switch(state){
163  case redraw_background:
164  app_time=0;
165  if (!draw.busy()){
166  data->update(RENDER,(int32_t)0);
167  if (animated_wallpaper.getNextFrameChunk()){
168  draw.bltSDAnimationFullScreen(&animated_wallpaper);
169  data->update(AM_REDRAW_BG,(int32_t)app_time);
170  } else Serial.println(F("M bad chunk"));
171  } else Serial.println(F("M draw busy"));
172  if ((data->read(RENDER) == 0) && animated_wallpaper.isFrameComplete()){
173  data->update(RENDER,(int32_t)1);
174  state = redraw_objects;
175  }
176  break;
177 
178  case redraw_wait://idle render time to give the screen refresh a head start
179  if (drt > 22){ //magic number is to tune the delay between frame buffer writes
180  //as the data is simutaniously being written out to SPI.
181  //This ensures we are not overwritting the current frame
182  //with the next frame.
183  if(exclusive_app_render) state = redraw_objects;
184  else state = redraw_background;
185  };
186  //note: this is a good place for application manager housekeeping tasks where screen access is not required
187  data->update(AM_AUDIO_CPU_MAX,(float32_t)AudioProcessorUsageMax());
188  float32_t cpu;
189  cpu = AudioProcessorUsage();
190  data->update(AM_AUDIO_CPU,(float32_t)cpu);
191  data->update(AM_AUDIO_MEM_MAX,(int32_t)AudioMemoryUsageMax());
192  data->update(AM_AUDIO_MEM,(int32_t)AudioMemoryUsage());
193  break;
194 
195  case redraw_render:
196  app_time=0;
197  data->update(RENDER_PERIOD,(int32_t)drt);
198  data->update(RENDER,(int32_t)4);
199  data->increment(RENDER_FRAME);
200  draw.updateScreenAsync(false);//updateScreenAsyncFrom(&draw,false);
201  state = redraw_wait;
202  display_refresh_time = 0;
203  break;
204 
205  case redraw_objects:
206  node = root;
207  if (exclusive_app_render == false){
208  do{
209  app_time=0;
210  if (node->et_render_period > node->render_period_max) node->render_period_max = node->et_render_period;
211  node->render(); //update active window
212  node->render_cpu_time = app_time;
213  if (node->render_cpu_time > node->render_cpu_time_max){
214  node->render_cpu_time_max = node->render_cpu_time;
215  }
216  //update the data dictionary
217  node->et_render_period =0;
218  node=node->next_app_node;//check next node
219  }while(node !=NULL);
220  };
221  data->update(RENDER,(int32_t)2);
222  data->increment(UPDATE_CALLS);
223  state = redraw_popup;
224  break;
225 
226  case redraw_popup:
227  app_time=0;
228  //draw any popups
229  if (!app_popup_stack_index==0 && app_popup_stack[app_popup_stack_index-1] != 0){
230  node = getApp(app_popup_stack[app_popup_stack_index-1]);
231  if (node != 0){
232  //Serial.println(F("AppManager::redraw_popup"));
233  node->render();
234  }
235  }
236  data->update(RENDER,(int32_t)3);
237  state = redraw_render;
238  break;
239  }
240  //always call the apps updateRT function on every loop
241  touch.update();
242  touch_updated = true;
243  data->increment(RT_CALLS);
244  node = root;
245  bool isactive_child;
246  do{
247  app_time=0;
248  if (node->update_priority > node->update_priority_counter){
249  node->update_priority_counter++;
250  }else{
251  node->update_priority_counter = 0;
252  if (node->et_update_period > node->update_period_max) node->update_period_max = node->et_update_period;
253  node->update(); //real time update (always called)
254  node->update_cpu_time = app_time;
255  if (node->update_cpu_time > node->update_cpu_time_max) node->update_cpu_time_max = node->update_cpu_time;
256  node->et_update_period =0;
257  }
258  isactive_child = false;
259  if (node->id == active_app_id && !draw.busy()) {
260  if (p_active_app != node){
261  if (p_active_app != 0) p_active_app->onFocusLost();
262  node->onFocus();
263  p_active_app = node;
264  }
265  }
266  if (node->parent_node!=NULL){if(node->parent_node->id == active_app_id){isactive_child = true;}}; //send event triggers to any child apps
267  if (node->id == active_app_id || isactive_child) {
268  //Serial.print("AppManager::updating active application");Serial.println(active_app_id);
269  //trigger any events and then call the update function for this 'active' app or child node
270  if (update_analog){
271  data->update(AM_AN1,(int32_t)analog.readAN1());
272  data->update(AM_AN2,(int32_t)analog.readAN2());
273  data->update(AM_AN3,(int32_t)analog.readAN3());
274  data->update(AM_AN4,(int32_t)analog.readAN4());
275 
276  node->onAnalog1(analog.readAN1(),analog.freadAN1());
277  node->onAnalog2(analog.readAN2(),analog.freadAN2());
278  node->onAnalog3(analog.readAN3(),analog.freadAN3());
279  node->onAnalog4(analog.readAN4(),analog.freadAN4());
280  }
281  if (touch_updated == true && touch.touched()) {
282  p = touch.getPoint();
283  //data->update("TOUCH_X",(int32_t)p.x);
284  //data->update("TOUCH_Y",(int32_t)p.y);
285  data->update("TOUCH_Z",(int32_t)p.z);
286  if (node->touch_state == 0){
287  node->onTouch(p.x, p.y);
288  node->touch_state=1;
289  } else{
290  node->onTouchDrag(p.x, p.y);
291  }
292  } else if (node->touch_state==1){
293  node->touch_state=0;
294  node->onTouchRelease(p.x, p.y);
295  }
296  }
297  node->cycle_time = cycle_time;
298  if (node->cycle_time > node->cycle_time_max){
299  node->cycle_time_max = node->cycle_time;
300  }
301  node=node->next_app_node;//check next node
302  }while(node !=NULL);
303 
304  if (cycle_time > cycle_time_max){
305  cycle_time_max = cycle_time;
306  } else{
307  //delayMicroseconds((cycle_time_max - cycle_time)/100);
308  }
309 
310  if(monitor_update){
311  monitor_dd_update_timer = 0;
312  data->update(AM_LOOP_TIME,(int32_t)cycle_time_max);
313  data->update(AM_SERIAL_AVAIL,(int32_t)Serial.availableForWrite());
314  //data->update("FRAME_PTR1",(int32_t)draw.getFrameAddress());
315  //data->update("FRAME_PTR2",(int32_t)render.getFrameAddress());
316  int32_t free_mem;
317  void* htop;
318  free_mem = 1000;
319  cycle_time_max*=0.999;
320  htop = malloc(1000);
321  while (htop){
322  memset(htop,0x5A,1000);
323  heapTop = (uint32_t) htop;
324  free(htop);
325  free_mem += 1000;
326  htop = malloc(free_mem);
327  }
328 
329  data->update(FREE_MEM,(int32_t)(free_mem - 1000));
330  data->update(FREE_HEAP,(int32_t)(0x20280000 - heapTop));
331  data->update(FREE_LOCAL,(int32_t)(0x2007F000 - (uint32_t)(&heapTop)));
332  data->update(FREE_STACK,(int32_t)FreeStack());
333 
334  heapTop = 0;
335  data->update(CPU_TEMP,(float32_t)tempmonGetTemp());
336  AudioProcessorUsageMaxReset();
337  AudioMemoryUsageMaxReset();
338  }
