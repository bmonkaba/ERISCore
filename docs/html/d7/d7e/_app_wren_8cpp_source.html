605 void FLASHMEM AppWren::MessageHandler(AppBaseClass *sender, const char *message){
606  if(sender->isName("SCI")){
607  if(0 == strncmp(message,"WREN_SCRIPT_EXECUTE",strlen("WREN_SCRIPT_EXECUTE"))) {
608  Serial.println(F("M AppWren::MessageHandler: WREN_SCRIPT_COMPILE -> compileOnly = false"));
609  compileOnly = false;
610  return;
611  }else if(0 == strncmp(message,"WREN_SCRIPT_COMPILE",strlen("WREN_SCRIPT_COMPILE"))) {
612  Serial.println(F("M AppWren::MessageHandler: WREN_SCRIPT_EXECUTE -> compileOnly = true"));
613  compileOnly = true;
614  return;
615  }else if(0 == strncmp(message,"WREN_SCRIPT_SAVE",strlen("WREN_SCRIPT_SAVE"))){
616  char cmd[128];
617  int total_read;
618  total_read = sscanf(message, "%127s %23s\n" , cmd, wren_module_name);
619  if (total_read==2){
620  strcat(wren_module_name,".wren");
621  Serial.print(F("M AppWren::MessageHandler: WREN_SCRIPT_SAVE as: "));
622  Serial.println(wren_module_name);
623  save_module = true;
624  }
625  return;
626  }else{ //if the message payload is not a command then assume its the data
627  if(save_module){
628  set_arm_clock(600000000);
629  FsFile file;
630  sd->chdir("/wren");
631  sd->remove(wren_module_name);
632  file = sd->open(wren_module_name, O_WRONLY|O_CREAT);
633  file.write(message,strlen(message));
634  file.close();
635  save_module = false;
636  Serial.println(F("M AppWren::MessageHandler: WREN_SCRIPT_SAVE complete"));
637  }else if(compileOnly){
638  Serial.println(F("M AppWren::MessageHandler: Compiling the received script"));
639  return;
640  }else{ //execute the script
641  Serial.println(F("M ************************************************************************"));
642  Serial.println(F("M AppWren::MessageHandler: Restarting the VM"));
643  if(!has_focus){
644  //cycling the focus triggers the parent app to recover any system changes made by the VM
645  getFocus();
646  returnFocus();
647  }else{
648  returnFocus();
649  }
650  restartVM();
651  Serial.println(F("M AppWren::MessageHandler: Loading the received script"));
652  loadScript(message);
653 
654 
655  Serial.println(F("M AppWren::MessageHandler: request complete"));
656  Serial.println(F("M ************************************************************************"));
657  return;
658  }
659  }
660  } else{
661  //All other messages from senders other than the SCI will be forwarded to the VM
662  //SerialUSB1 is dedicated to serial comms with the VM for stdio/stderr
663  wrenEnsureSlots(vm, 3);
664  wrenSetSlotHandle(vm, 0, h_slot0);//App
665  wrenSetSlotString(vm, 1, sender->name);//sender
666  wrenSetSlotString(vm, 2, message);//message
667  isWrenResultOK(wrenCall(vm,h_messageHandler));
668  }
669 }

671 bool FLASHMEM AppWren::dynamicSurfaceManager(){

675  surface_cache = new Surface((uint16_t *)draw->getFrameAddress(),128,128);

677  }else{
678  surface_cache = new Surface(surface_mempool, widget_width, widget_height);
679  memset(surface_mempool,0,sizeof(uint16_t)*widget_width*widget_height);
680  }
681  }else{
682  //check for state changes and adjust the buffer accordingly
683  if (has_pop || has_focus){
684  if( (width != surface_cache->getWidth()) || (height != surface_cache->getHeight()) ){
685  delete(surface_cache);
686  surface_cache = new Surface(surface_mempool, width, height);
687  memset(surface_mempool,0,sizeof(uint16_t)*width*height);
688  }
689  }else{
690  if( (widget_width != surface_cache->getWidth()) || (widget_height != surface_cache->getHeight()) ){
691  delete(surface_cache);
692  surface_cache = new Surface(surface_mempool, widget_width, widget_height);
693  memset(surface_mempool,0,sizeof(uint16_t)*widget_width*widget_height);
694  }
695  }
696  }
697  if (surface_cache==NULL)return false;
698  wrenCollectGarbage(vm);
699  return true;

800 const char * FLASHMEM AppWren::loadModuleSource(const char * name){

803  int16_t file_size; //max file size 32K

805  strcat(file_name,".wren");
806  sd->chdir("/wren");
807  file = sd->open(file_name, O_RDONLY);
808  file_size = file.available() + 1;
809 #ifdef USE_EXTMEM
810  module_load_buffer = (char*)extmem_malloc(file_size);
811 #else
812  module_load_buffer = (char*)malloc(file_size);

815  Serial.println(F("M AppWren::loadModuleSource ERROR: Not Enough Memory"));
816  return NULL;
817  }
818  memset(module_load_buffer,0,file_size);
819  file.read(module_load_buffer,file_size);
820  file.close();
821  return module_load_buffer;
822 }
