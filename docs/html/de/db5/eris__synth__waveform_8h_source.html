39 class AudioSynthWaveform2k : public AudioStream
40 {
41 public:
42  AudioSynthWaveform2k(void) : AudioStream(0,NULL),
43  phase_accumulator(0), phase_increment(0), phase_offset(0),
44  magnitude(0), pulse_width(0x40000000),
45  arbdata(NULL), sample(0), tone_type(WAVEFORM_SINE),
46  tone_offset(0) {
47  }
48 
49  void frequency(float freq) {
50  if (freq < 0.0f) {
51  freq = 0.0;
52  } else if (freq > AUDIO_SAMPLE_RATE_EXACT / 2.0f) {
53  freq = AUDIO_SAMPLE_RATE_EXACT / 2.0f;
54  }
55  phase_increment = freq * (4294967296.0f / AUDIO_SAMPLE_RATE_EXACT);
56  if (phase_increment > 0x7FFE0000u) phase_increment = 0x7FFE0000;
57  }
58  void phase(float angle) {
59  if (angle < 0.0f) {
60  angle = 0.0;
61  } else if (angle > 360.0f) {
62  angle = angle - 360.0f;
63  if (angle >= 360.0f) return;
64  }
65  phase_offset = angle * (float)(4294967296.0 / 360.0);
66  }
67  //get phase added for Eris Core
68  float getPhase() {
69  return phase_offset/(4294967296.0 / 360.0);
70  }
71  void amplitude(float n) { // 0 to 1.0
72  if (n < 0) {
73  n = 0;
74  } else if (n > 1.0f) {
75  n = 1.0;
76  }
77  magnitude = n * 65536.0f;
78  }
79  void offset(float n) {
80  if (n < -1.0f) {
81  n = -1.0f;
82  } else if (n > 1.0f) {
83  n = 1.0f;
84  }
85  tone_offset = n * 32767.0f;
86  }
87  void pulseWidth(float n) { // 0.0 to 1.0
88  if (n < 0) {
89  n = 0;
90  } else if (n > 1.0f) {
91  n = 1.0f;
92  }
93  pulse_width = n * 4294967296.0f;
94  }
95  void begin(short t_type) {
96  phase_offset = 0;
97  tone_type = t_type;
98  if (t_type == WAVEFORM_BANDLIMIT_SQUARE)
99  band_limit_waveform.init_square (phase_increment) ;
100  else if (t_type == WAVEFORM_BANDLIMIT_PULSE)
101  band_limit_waveform.init_pulse (phase_increment, pulse_width) ;
102  else if (t_type == WAVEFORM_BANDLIMIT_SAWTOOTH || t_type == WAVEFORM_BANDLIMIT_SAWTOOTH_REVERSE)
103  band_limit_waveform.init_sawtooth (phase_increment) ;
104  }
105  void begin(float t_amp, float t_freq, short t_type) {
106  amplitude(t_amp);
107  frequency(t_freq);
108  phase_offset = 0;
109  begin (t_type);
110  }
111  //added for Eris Core
112  void arbitraryProgram(uint8_t program);
113  void arbitraryWaveform(const int16_t *data, float maxFreq) {
114  arbdata = data;
115  }
116  virtual void update(void);
117 
118 private:
119  uint32_t phase_accumulator;
120  uint32_t phase_increment;
121  uint32_t phase_offset;
122  int32_t magnitude;
123  uint32_t pulse_width;
124  const int16_t *arbdata;
125  int16_t sample; // for WAVEFORM_SAMPLE_HOLD
126  short tone_type;
127  int16_t tone_offset;
128  BandLimitedWaveform band_limit_waveform ;
129 };
