51 class AudioSynthWaveform : public AudioStream
52 {
53 public:
54  AudioSynthWaveform(void) : AudioStream(0,NULL),
55  phase_accumulator(0), phase_increment(0), phase_offset(0),
56  magnitude(0), pulse_width(0x40000000),
57  arbdata(NULL), sample(0), tone_type(WAVEFORM_SINE),
58  tone_offset(0) {
59  }
60 
61  void frequency(float freq) {
62  if (freq < 0.0) {
63  freq = 0.0;
64  } else if (freq > AUDIO_SAMPLE_RATE_EXACT / 2) {
65  freq = AUDIO_SAMPLE_RATE_EXACT / 2;
66  }
67  phase_increment = freq * (4294967296.0 / AUDIO_SAMPLE_RATE_EXACT);
68  if (phase_increment > 0x7FFE0000u) phase_increment = 0x7FFE0000;
69  }
70  void phase(float angle) {
71  if (angle < 0.0) {
72  angle = 0.0;
73  } else if (angle > 360.0) {
74  angle = angle - 360.0;
75  if (angle >= 360.0) return;
76  }
77  phase_offset = angle * (4294967296.0 / 360.0);
78  }
79  float getPhase() {
80  return phase_offset/(4294967296.0 / 360.0);
81  }
82  void amplitude(float n) { // 0 to 1.0
83  if (n < 0) {
84  n = 0;
85  } else if (n > 1.0) {
86  n = 1.0;
87  }
88  magnitude = n * 65536.0;
89  }
90  void offset(float n) {
91  if (n < -1.0) {
92  n = -1.0;
93  } else if (n > 1.0) {
94  n = 1.0;
95  }
96  tone_offset = n * 32767.0;
97  }
98  void pulseWidth(float n) { // 0.0 to 1.0
99  if (n < 0) {
100  n = 0;
101  } else if (n > 1.0) {
102  n = 1.0;
103  }
104  pulse_width = n * 4294967296.0;
105  }
106  void begin(short t_type) {
107  phase_offset = 0;
108  tone_type = t_type;
109  }
110  void begin(float t_amp, float t_freq, short t_type) {
111  amplitude(t_amp);
112  frequency(t_freq);
113  phase_offset = 0;
114  tone_type = t_type;
115  }
116  void arbitraryWaveform(const int16_t *data, float maxFreq) {
117  arbdata = data;
118  }
119  virtual void update(void);
120 
121 public:
122  uint32_t phase_accumulator;
123  uint32_t phase_increment;
124  uint32_t phase_offset;
125  int32_t magnitude;
126  uint32_t pulse_width;
127  const int16_t *arbdata;
128  int16_t sample; // for WAVEFORM_SAMPLE_HOLD
129  short tone_type;

131 };
